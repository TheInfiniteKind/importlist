001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.util.Comparator;
004    
005    import javax.swing.DefaultRowSorter;
006    import javax.swing.table.TableModel;
007    
008    import org.apache.commons.lang.Validate;
009    
010    import com.moneydance.modules.features.importlist.util.AlphanumComparator;
011    import com.moneydance.modules.features.importlist.util.Preferences;
012    
013    /**
014     * An implementation of <code>RowSorter</code> that provides sorting for a
015     * <code>ListTableModel</code>.
016     *
017     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
018     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
019     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
020     */
021    class FileTableRowSorter extends DefaultRowSorter<TableModel, Integer> {
022    
023        private final Preferences    prefs;
024        private final ListTableModel listTableModel;
025        private final Comparator<?> comparator;
026    
027        /**
028         * @param argListTableModel The table model that is to be sorted
029         */
030        FileTableRowSorter(final ListTableModel argListTableModel) {
031            super();
032            Validate.notNull(argListTableModel, "argListTableModel can't be null");
033            this.prefs          = Preferences.getInstance();
034            this.listTableModel = argListTableModel;
035            this.comparator     = new AlphanumComparator();
036            this.setModelWrapper(new FileModelWrapper());
037        }
038    
039        @Override
040        public final boolean isSortable(final int column) {
041            String columnName = this.listTableModel.getColumnName(column);
042            return this.prefs.getDescName().equals(columnName)
043            || this.prefs.getDescModified().equals(columnName);
044        }
045    
046        @Override
047        public Comparator<?> getComparator(final int column) {
048            return this.comparator;
049        }
050    
051        private class FileModelWrapper
052        extends DefaultRowSorter.ModelWrapper<TableModel, Integer> {
053    
054            @Override
055            public final int getColumnCount() {
056                return FileTableRowSorter.this.listTableModel.getColumnCount();
057            }
058    
059            @Override
060            public final Integer getIdentifier(final int columnIndex) {
061                return columnIndex;
062            }
063    
064            @Override
065            public final TableModel getModel() {
066                return FileTableRowSorter.this.listTableModel;
067            }
068    
069            @Override
070            public final int getRowCount() {
071                return FileTableRowSorter.this.listTableModel.getRowCount();
072            }
073    
074            @Override
075            public final Object getValueAt(final int row, final int column) {
076                ListTableModel model = FileTableRowSorter.this.listTableModel;
077                String columnName    = model.getColumnName(column);
078    
079                if (FileTableRowSorter.this.prefs.getDescModified().equals(
080                        columnName)) {
081                    return model.getFileAt(row).lastModified() + "";
082                } else {
083                    return model.getValueAt(row, column);
084                }
085            }
086        }
087    }
