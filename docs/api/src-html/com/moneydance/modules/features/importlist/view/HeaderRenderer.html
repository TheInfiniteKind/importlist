001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.awt.Component;
004    
005    import javax.swing.JComponent;
006    import javax.swing.JLabel;
007    import javax.swing.JTable;
008    import javax.swing.SwingConstants;
009    import javax.swing.table.TableCellRenderer;
010    
011    /**
012     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
013     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
014     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
015     */
016    public class HeaderRenderer implements TableCellRenderer {
017    
018        private static final long serialVersionUID = 3121884943197710031L;
019        private final TableCellRenderer defaultHeaderTableCellRenderer;
020    
021        HeaderRenderer(
022                final TableCellRenderer argDefaultHeaderTableCellRenderer) {
023            this.defaultHeaderTableCellRenderer = argDefaultHeaderTableCellRenderer;
024        }
025    
026        public final Component getTableCellRendererComponent(
027                final JTable table,
028                final Object value,
029                final boolean isSelected,
030                final boolean hasFocus,
031                final int row,
032                final int column) {
033            Component component =
034                this.defaultHeaderTableCellRenderer.getTableCellRendererComponent(
035                table, value, hasFocus, hasFocus, row, column);
036            ColorSchemeHelper.applyColorScheme(component, row);
037    
038            if (component instanceof JComponent) {
039                JComponent jComponent = (JComponent) component;
040                jComponent.setBorder(null);
041            }
042    
043            if (component instanceof JLabel) {
044                JLabel jLabel = (JLabel) component;
045                jLabel.setHorizontalAlignment(SwingConstants.LEFT);
046            }
047            return component;
048        }
049    }
