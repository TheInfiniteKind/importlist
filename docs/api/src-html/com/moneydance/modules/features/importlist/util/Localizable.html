017package com.moneydance.modules.features.importlist.util;
018
019import java.util.Locale;
020import java.util.Map;
021import java.util.ResourceBundle;
022import java.util.concurrent.ConcurrentHashMap;
023
024import org.apache.commons.lang3.text.StrSubstitutor;
025import org.apache.commons.lang3.text.WordUtils;
026
027/**
028 * This i18n class provides language-dependent strings such as labels and
029 * messages.
030 *
031 * @author Florian J. Breunig
032 */
033public final class Localizable {
034
035    private final ResourceBundle resourceBundle;
036
037    Localizable(final String localizableResource, final Locale locale) {
038        this.resourceBundle = ResourceBundle.getBundle(
039                localizableResource,
040                locale);
041    }
042
043    /**
044     * @return The title of the dialog that is displayed to choose the base
045     * directory.
046     */
047    public String getDirectoryChooserTitle() {
048        return this.resourceBundle.getString("directory_chooser_title");
049    }
050
051    /**
052     * @return Header of the "name" column.
053     */
054    public String getHeaderValueName() {
055        final String headerValueName =
056                this.resourceBundle.getString("header_value_name");
057        return Helper.INSTANCE.getSettings().getIndentationPrefix()
058                + headerValueName;
059    }
060
061    /**
062     * @return Header of the "modified" column.
063     */
064    public String getHeaderValueModified() {
065        final String headerValueModified =
066                this.resourceBundle.getString("header_value_modified");
067        return String.format("%s%s",
068                Helper.INSTANCE.getSettings().getIndentationPrefix(),
069                headerValueModified);
070    }
071
072    /**
073     * @return Header of the "import" column.
074     */
075    public String getHeaderValueImport() {
076        final String headerValueImport =
077                this.resourceBundle.getString("header_value_import");
078        return String.format("%s%s",
079                Helper.INSTANCE.getSettings().getIndentationPrefix(),
080                headerValueImport);
081    }
082
083    /**
084     * @return Header of the "delete" column.
085     */
086    public String getHeaderValueDelete() {
087        final String headerValueDelete =
088                this.resourceBundle.getString("header_value_delete");
089        return String.format("%s%s",
090                Helper.INSTANCE.getSettings().getIndentationPrefix(),
091                headerValueDelete);
092    }
093
094    /**
095     * @return Label of the "import" button.
096     */
097    public String getLabelImportOneButton() {
098        return this.resourceBundle.getString("label_import_one_button");
099    }
100
101    /**
102     * @return Label of the "import all" button.
103     */
104    public String getLabelImportAllButton() {
105        return this.resourceBundle.getString("label_import_all_button");
106    }
107
108    /**
109     * @return Label of the "delete" button.
110     */
111    public String getLabelDeleteOneButton() {
112        return this.resourceBundle.getString("label_delete_one_button");
113    }
114
115    /**
116     * @return Label of the "delete all" button.
117     */
118    public String getLabelDeleteAllButton() {
119        return this.resourceBundle.getString("label_delete_all_button");
120    }
121
122    /**
123     * @param baseDirectory The base directory which could not be opened.
124     * @return Error message to be displayed if the base directory could not be
125     * opened.
126     */
127    public String getErrorMessageBaseDirectory(final String baseDirectory) {
128        final String templateString = this.resourceBundle.getString(
129                "error_message_base_directory");
130
131        Map<String, String> valuesMap =
132                new ConcurrentHashMap<String, String>(1);
133        valuesMap.put("import.dir",  getMarkupFilename(baseDirectory));
134        StrSubstitutor sub = new StrSubstitutor(valuesMap);
135
136        return sub.replace(templateString);
137    }
138
139    /**
140     * @param filename The name of the file that is to be deleted.
141     * @return Confirmation message to be displayed before a file will be
142     *  deleted.
143     */
144    public String getConfirmationMessageDeleteOneFile(final String filename) {
145        final String templateString = this.resourceBundle.getString(
146                "confirmation_message_delete_one_file");
147
148        Map<String, String> valuesMap =
149                new ConcurrentHashMap<String, String>(1);
150        valuesMap.put("filename",  getMarkupFilename(filename));
151        StrSubstitutor sub = new StrSubstitutor(valuesMap);
152
153        return sub.replace(templateString);
154    }
155
156    /**
157     * @param size The number of all files that are to be deleted.
158     * @return Confirmation message to be displayed before all files will be
159     *  deleted.
160     */
161    public String getConfirmationMessageDeleteAllFiles(final int size) {
162        final String templateString = this.resourceBundle.getString(
163                "confirmation_message_delete_all_files");
164
165        Map<String, String> valuesMap =
166                new ConcurrentHashMap<String, String>(1);
167        valuesMap.put("no.files",  String.valueOf(size));
168        StrSubstitutor sub = new StrSubstitutor(valuesMap);
169
170        return sub.replace(templateString);
171    }
172
173    /**
174     * @param filename The name of the file that cannot be deleted.
175     * @return Error message to be displayed if a file cannot be deleted.
176     */
177    public String getErrorMessageDeleteFile(final String filename) {
178        final String templateString = this.resourceBundle.getString(
179                "error_message_delete_file");
180
181        Map<String, String> valuesMap =
182                new ConcurrentHashMap<String, String>(1);
183        valuesMap.put("filename",  getMarkupFilename(filename));
184        StrSubstitutor sub = new StrSubstitutor(valuesMap);
185
186        return sub.replace(templateString);
187    }
188
189    /**
190     * @return Delete button in confirmation message.
191     */
192    public String getOptionDeleteFile() {
193        return this.resourceBundle.getString("option_delete_file");
194    }
195
196    /**
197     * @return Cancel button in confirmation message.
198     */
199    public String getOptionCancel() {
200        return this.resourceBundle.getString("option_cancel");
201    }
202
203    /**
204     * @param baseDirectory The base directory which does not contain any files
205     * to display.
206     * @return Message to display if the list of files is empty.
207     */
208    public String getEmptyMessage(final String baseDirectory) {
209        final String templateString = this.resourceBundle.getString(
210                "empty_message");
211
212        Map<String, String> valuesMap =
213                new ConcurrentHashMap<String, String>(1);
214        valuesMap.put("import.dir", baseDirectory);
215        StrSubstitutor sub = new StrSubstitutor(valuesMap);
216
217        return sub.replace(templateString);
218    }
219
220    private static String getMarkupFilename(final String filename) {
221        return WordUtils.wrap(
222                filename,
223                Helper.INSTANCE.getSettings().getMessageFilenameLineMaxLength(),
224                "<br>", /*wrapLongWords*/ true);
225    }
226}
