017package com.moneydance.modules.features.importlist.presentation;
018
019import com.moneydance.apps.md.view.gui.MoneydanceLAF;
020
021import javax.swing.JComponent;
022import javax.swing.JSplitPane;
023import javax.swing.border.EmptyBorder;
024
025/**
026 * @author Florian J. Breunig
027 */
028public final class SplitPaneFactory implements ComponentFactory {
029
030    private       JSplitPane splitPane;
031    private final ComponentFactory topComponentFactory;
032    private final ComponentFactory bottomComponentFactory;
033
034    public SplitPaneFactory(
035            final ComponentFactory argTopComponentFactory,
036            final ComponentFactory argBottomComponentFactory) {
037
038        this.topComponentFactory    = argTopComponentFactory;
039        this.bottomComponentFactory = argBottomComponentFactory;
040    }
041
042    private void init() {
043        this.splitPane = new JCustomSplitPane(JSplitPane.VERTICAL_SPLIT);
044        this.splitPane.setResizeWeight(1.0);
045        this.splitPane.setDividerSize(0);
046        this.splitPane.setContinuousLayout(true);
047        this.splitPane.setOpaque(false);
048        this.splitPane.setBorder(MoneydanceLAF.homePageBorder);
049    }
050
051    @Override
052    public JSplitPane getComponent() {
053        if (this.splitPane == null) {
054            this.init();
055        }
056
057        JComponent topComponent = this.topComponentFactory.getComponent();
058        topComponent.setBorder(new EmptyBorder(0, 0, 0, 0));
059        this.splitPane.setTopComponent(topComponent);
060
061        JComponent bottomComponent = this.bottomComponentFactory.getComponent();
062        this.splitPane.setBottomComponent(bottomComponent);
063
064        return this.splitPane;
065    }
066}
