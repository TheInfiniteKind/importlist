001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.awt.Color;
004    import java.awt.Component;
005    
006    import javax.swing.JComponent;
007    
008    /**
009     * This helper class applies a given color scheme to a <code>Component</code>.
010     * It is used by different <code>TableCellRenderer</code>s to guarantee a
011     * consistent style in a table.
012     *
013     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
014     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
015     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
016     */
017    final class ColorSchemeHelper {
018    
019        private static final long serialVersionUID = -2724848115953702911L;
020        private static Color foreground;
021        private static Color background;
022        private static Color backgroundAlt;
023    
024        /**
025         * Private constructor prevents this class from being instantiated.
026         */
027        private ColorSchemeHelper() {
028        }
029    
030        static void applyColorScheme(
031                final Component component,
032                final int row) {
033            component.setForeground(foreground);
034            component.setBackground(background);
035            if (row % 2 == 0) {
036                if (component instanceof JComponent) {
037                    JComponent jComponent = (JComponent) component;
038                    jComponent.setOpaque(true);
039                }
040                component.setBackground(backgroundAlt);
041            }
042        }
043    
044        static void setForeground(final Color argForeground) {
045            foreground = argForeground;
046        }
047    
048        static void setBackground(final Color argBackground) {
049            background = argBackground;
050        }
051    
052        static void setBackgroundAlt(final Color argBackgroundAlt) {
053            backgroundAlt = argBackgroundAlt;
054        }
055    }
