032    /**
033     * The user interface that is displayed on the homepage.
034     *
035     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
036     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
037     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
038     */
039    public class View implements HomePageView, Runnable {
040    
041        private final FileAdministration    fileAdministration;
042        private final ListTableModel        listTableModel;
043        private final JTable                table;
044        private final ColumnFactory         columnFactory;
045        private final JScrollPane           scrollPane;
046    
047    
048        public View(final FileAdministration argFileAdministration) {
049            Validate.notNull(argFileAdministration,
050            "argFileAdministration can't be null");
051            this.fileAdministration = argFileAdministration;
052    
053            this.listTableModel = new ListTableModel(
054                    this.fileAdministration.getFiles());
055    
056            this.table = new JTable(this.listTableModel);
057            this.table.setShowGrid(false);
058            this.table.setShowVerticalLines(false);
059            this.table.setShowHorizontalLines(false);
060            this.table.setIntercellSpacing(new Dimension(0, 0));
061            this.table.setColumnSelectionAllowed(false);
062            this.table.setRowSelectionAllowed(false);
063            this.table.setCellSelectionEnabled(false);
064    
065            Preferences prefs            = Preferences.getInstance();
066            JTableHeader tableHeader     = this.table.getTableHeader();
067            TableColumnModel columnModel = this.table.getColumnModel();
068    
069            this.columnFactory = new ColumnFactory(
070                    this.fileAdministration,
071                    tableHeader.getDefaultRenderer());
072    
073            TableColumn nameCol = this.table.getColumn(Constants.DESC_NAME);
074            int nameColNo = columnModel.getColumnIndex(Constants.DESC_NAME);
075            nameCol.setIdentifier(Constants.DESC_NAME);
076            nameCol.setHeaderValue(Constants.HEADER_VALUE_NAME);
077            nameCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
078            nameCol.setCellRenderer(this.columnFactory.getLabelRenderer());
079            nameCol.setMinWidth(Constants.MIN_COLUMN_WIDTH);
080            nameCol.setPreferredWidth(prefs.getColumnWidths(nameColNo));
081    
082            TableColumn modifiedCol = this.table.getColumn(Constants.DESC_MODIFIED);
083            int modifiedColNo = columnModel.getColumnIndex(Constants.DESC_MODIFIED);
084            modifiedCol.setIdentifier(Constants.DESC_MODIFIED);
085            modifiedCol.setHeaderValue(Constants.HEADER_VALUE_MODIFIED);
086            modifiedCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
087            modifiedCol.setCellRenderer(this.columnFactory.getLabelRenderer());
088            modifiedCol.setMinWidth(Constants.MIN_COLUMN_WIDTH);
089            modifiedCol.setPreferredWidth(prefs.getColumnWidths(modifiedColNo));
090    
091            TableColumn importCol = this.table.getColumn(Constants.DESC_IMPORT);
092            int importColNo = columnModel.getColumnIndex(Constants.DESC_IMPORT);
093            importCol.setIdentifier(Constants.DESC_IMPORT);
094            importCol.setHeaderValue(Constants.HEADER_VALUE_IMPORT);
095            importCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
096            importCol.setCellRenderer(this.columnFactory.getButtonRenderer());
097            importCol.setCellEditor(this.columnFactory.getImportEditor());
098            importCol.setResizable(Constants.BUTTON_RESIZABLE);
099            importCol.setMinWidth(Constants.MIN_COLUMN_WIDTH);
100            importCol.setPreferredWidth(prefs.getColumnWidths(importColNo));
101    
102            TableColumn deleteCol = this.table.getColumn(Constants.DESC_DELETE);
103            int deleteColNo = columnModel.getColumnIndex(Constants.DESC_DELETE);
104            deleteCol.setIdentifier(Constants.DESC_DELETE);
105            deleteCol.setHeaderValue(Constants.HEADER_VALUE_DELETE);
106            deleteCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
107            deleteCol.setCellRenderer(this.columnFactory.getButtonRenderer());
108            deleteCol.setCellEditor(this.columnFactory.getDeleteEditor());
109            deleteCol.setResizable(Constants.BUTTON_RESIZABLE);
110            deleteCol.setMinWidth(Constants.MIN_COLUMN_WIDTH);
111            deleteCol.setPreferredWidth(prefs.getColumnWidths(deleteColNo));
112    
113            // resizing the columns
114            TableListener tableListener =
115                new TableListener(this.table);
116            columnModel.addColumnModelListener(tableListener);
117    
118            // reordering the columns
119            tableHeader.setReorderingAllowed(Constants.ALLOW_REORDERING);
120    
121            // sorting the columns
122            DefaultRowSorter<TableModel, Integer> rowSorter =
123                new FileTableRowSorter(this.listTableModel);
124            List<RowSorter.SortKey> sortKeys = new ArrayList<RowSorter.SortKey>();
125            sortKeys.add(Preferences.getInstance().getSortKey());
126            rowSorter.setSortKeys(sortKeys);
127    
128            rowSorter.addRowSorterListener(tableListener);
129    
130            Comparator<String> comparator = new AlphanumComparator();
131            rowSorter.setComparator(nameColNo, comparator);
132            this.table.setRowSorter(rowSorter);
133    
134            this.table.setPreferredScrollableViewportSize(
135                    new Dimension(
136                            Constants.PREFERRED_TABLE_WIDTH,
137                            Constants.PREFERRED_TABLE_HEIGHT));
138    
139            this.scrollPane = new JScrollPane();
140            //see moneydance.com/pipermail/moneydance-dev/2006-September/000075.html
141            try {
142                this.scrollPane.setBorder(MoneydanceLAF.homePageBorder);
143            } catch (Throwable e) {
144                e.printStackTrace(System.err);
145            }
146        }
147    
148    
149        @Override
150        public final void refresh() {
151            Preferences prefs = Preferences.getInstance();
152            this.listTableModel.setDateFormatter(prefs.getDateFormatter());
153            this.listTableModel.setTimeFormatter(prefs.getTimeFormatter());
154            this.columnFactory.setForegroundColor(prefs.getForegroundColor());
155            this.columnFactory.setBackgroundColor(prefs.getBackgroundColor());
156            this.columnFactory.setBackgroundColorAlt(prefs.getBackgroundColorAlt());
157            this.table.setRowHeight(prefs.getBodyRowHeight());
158            this.table.getTableHeader().setFont(prefs.getHeaderFont());
159            this.table.getTableHeader().setSize(
160                    this.table.getTableHeader().getWidth(),
161                    prefs.getHeaderRowHeight());
162            this.scrollPane.setBackground(prefs.getBackgroundColor());
163            this.scrollPane.getViewport().setBackground(prefs.getBackgroundColor());
164    
165            if (this.fileAdministration.isDirty() && this.scrollPane.isVisible()) {
166                this.fileAdministration.setDirty(false);
167                try {
168                    if (SwingUtilities.isEventDispatchThread()) {
169                        this.run();
170                    } else {
171                        SwingUtilities.invokeAndWait(this);
172                    }
173                } catch (InterruptedException e) {
174                    e.printStackTrace(System.err);
175                } catch (InvocationTargetException e) {
176                    e.printStackTrace(System.err);
177                }
178            }
179        }
180    
181    
182        @Override
183        public final void run() {
184            this.fileAdministration.reloadFiles();
185            this.listTableModel.fireTableDataChanged();
186    
187            if (this.listTableModel.getRowCount() == 0) {
188                String message = "<html><center><em>There are currently no files "
189                    + "to import<br>in " + this.fileAdministration.getImportDir()
190                    + "</em><center></html>";
191                Preferences preferences = Preferences.getInstance();
192                JLabel textPanel = new JLabel(message);
193                textPanel.setHorizontalAlignment(SwingConstants.CENTER);
194                textPanel.setBackground(preferences.getBackgroundColor());
195                this.scrollPane.setViewportView(textPanel);
196                this.scrollPane.setPreferredSize(
197                        new Dimension(
198                                Constants.PREFERRED_EMPTY_MESSAGE_WIDTH,
199                                Constants.PREFERRED_EMPTY_MESSAGE_HEIGHT));
200            } else {
201                this.scrollPane.setViewportView(this.table);
202                this.scrollPane.setPreferredSize(
203                        new Dimension(
204                                Constants.PREFERRED_TABLE_WIDTH,
205                                Constants.PREFERRED_TABLE_HEIGHT));
206            }
207        }
208    
209    
210        @Override
211        public final JComponent getGUIView(final RootAccount rootAccount) {
212            return this.scrollPane;
213        }
214    
215    
216        @Override
217        public final void reset() {
218        }
219    
220    
221        @Override
222        public final void setActive(final boolean active) {
223            this.scrollPane.setVisible(active);
224            this.refresh();
225        }
226    
227    
228        @Override
229        public final String getID() {
230            return Constants.ID;
231        }
232    
233    
234        @Override
235        public final String toString() {
236            return Constants.EXTENSION_NAME;
237        }
