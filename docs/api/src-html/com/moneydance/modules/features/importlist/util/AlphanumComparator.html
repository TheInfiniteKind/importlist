022package com.moneydance.modules.features.importlist.util;
023
024import java.util.Comparator;
025
026/**
027 * This is an updated version of the
028 * <a href="http://www.davekoelle.com/alphanum.html">Alphanum Algorithm</a>.
029 *
030 * @author Daniel Migowski, Andre Bogus, David Koelle, and Florian J. Breunig.
031 */
032public enum AlphanumComparator implements Comparator<String> {
033
034    /**
035     * Alphanum comparator instance.
036     */
037    ALPHANUM;
038
039    /**
040     * Serialization.
041     */
042    private static final long serialVersionUID = 1L;
043
044    /**
045     * The lowest ASCII symbol representing a digit.
046     */
047    private static final int  DIGIT_LOWER_BOUND = 48;
048
049    /**
050     * The highest ASCII symbol representing a digit.
051     */
052    private static final int  DIGIT_UPPER_BOUND = 57;
053
054    /**
055     * Restrictive constructor.
056     */
057    private AlphanumComparator() {
058        // Prevents this class from being instantiated from the outside.
059    }
060
061    /**
062     * @param character The character
063     * @return <code>true</code> if the given char represents a digit;
064     * <code>false</code> otherwise.
065     */
066    private static boolean isDigit(final char character) {
067        return character >= DIGIT_LOWER_BOUND
068                && character <= DIGIT_UPPER_BOUND;
069    }
070
071    /**
072     * The length of the string is passed in for improved efficiency
073     * (only need to calculate it once).
074     * @param string The string to be analyzed.
075     * @param slength The length of the string to be analyzed.
076     * @param marker The index of the string to start analyzing from.
077     * @return A substring containing only digits or only non-digits.
078     */
079    private static String getChunk(
080            final String string,
081            final int slength,
082            final int marker) {
083        int index = marker;
084        final StringBuilder chunk = new StringBuilder();
085        char character = string.charAt(index);
086        chunk.append(character);
087        index++;
088        if (isDigit(character)) {
089            while (index < slength) {
090                character = string.charAt(index);
091                if (!isDigit(character)) {
092                    break;
093                }
094                chunk.append(character);
095                index++;
096            }
097        } else {
098            while (index < slength) {
099                character = string.charAt(index);
100                if (isDigit(character)) {
101                    break;
102                }
103                chunk.append(character);
104                index++;
105            }
106        }
107        return chunk.toString();
108    }
109
110    /**
111     * @param string1 The first string to be compared.
112     * @param string2 The second string to be compared.
113     * @return A negative integer, zero, or a positive integer as the first
114     * argument is less than, equal to, or greater than the second.
115     */
116    @Override
117    public int compare(final String string1, final String string2) {
118        int thisMarker = 0;
119        int thatMarker = 0;
120        final int s1Length = string1.length();
121        final int s2Length = string2.length();
122
123        while (thisMarker < s1Length && thatMarker < s2Length) {
124            final String thisChunk
125            = getChunk(string1, s1Length, thisMarker);
126            thisMarker += thisChunk.length();
127
128            final String thatChunk
129            = getChunk(string2, s2Length, thatMarker);
130            thatMarker += thatChunk.length();
131
132            // If both chunks contain numeric characters, sort them numerically
133            int result = 0;
134            if (isDigit(thisChunk.charAt(0))
135                    && isDigit(thatChunk.charAt(0))) {
136                // Simple chunk comparison by length.
137                final int thisChunkLength = thisChunk.length();
138                result = thisChunkLength - thatChunk.length();
139                // If equal, the first different number counts
140                if (result == 0) {
141                    for (int i = 0; i < thisChunkLength; i++) {
142                        result = thisChunk.charAt(i) - thatChunk.charAt(i);
143                        if (result != 0) {
144                            return result;
145                        }
146                    }
147                }
148            } else {
149                result = thisChunk.compareTo(thatChunk);
150            }
151
152            if (result != 0) {
153                return result;
154            }
155        }
156
157        return s1Length - s2Length;
158    }
159}
