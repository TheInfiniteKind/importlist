017package com.moneydance.modules.features.importlist.presentation;
018
019import com.moneydance.modules.features.importlist.io.FileAdmin;
020import com.moneydance.modules.features.importlist.table.ColumnFactory;
021import com.moneydance.modules.features.importlist.util.Helper;
022import com.moneydance.modules.features.importlist.util.Localizable;
023import com.moneydance.modules.features.importlist.util.Preferences;
024import com.moneydance.modules.features.importlist.util.Settings;
025
026import java.awt.Dimension;
027import java.util.ArrayList;
028import java.util.List;
029
030import javax.swing.JScrollPane;
031import javax.swing.JTable;
032import javax.swing.RowSorter;
033import javax.swing.table.JTableHeader;
034import javax.swing.table.TableColumn;
035import javax.swing.table.TableColumnModel;
036import javax.swing.table.TableModel;
037
038/**
039 * @author Florian J. Breunig
040 */
041public final class BaseTableFactory implements ComponentFactory {
042
043    private final Settings        settings;
044    private final Preferences     prefs;
045    private final Localizable     localizable;
046    private final TableModel      tableModel;
047    private final JTable          table;
048    private final ColumnFactory   columnFactory;
049    private final JScrollPane     scrollPane;
050
051    public BaseTableFactory(
052            final TableModel argTableModel,
053            final FileAdmin argFileAdmin) {
054        this.tableModel = argTableModel;
055        this.settings = Helper.INSTANCE.getSettings();
056        this.prefs = Helper.INSTANCE.getPreferences();
057        this.localizable = Helper.INSTANCE.getLocalizable();
058        this.table = new JTable(this.tableModel);
059        this.table.setOpaque(false);
060        this.table.setShowGrid(false);
061        this.table.setShowVerticalLines(false);
062        this.table.setShowHorizontalLines(false);
063        this.table.setIntercellSpacing(new Dimension(0, 0));
064        this.table.setColumnSelectionAllowed(false);
065        this.table.setRowSelectionAllowed(false);
066        this.table.setCellSelectionEnabled(false);
067        JTableHeader tableHeader         = this.table.getTableHeader();
068        tableHeader.setOpaque(false);
069        TableColumnModel baseColumnModel = this.table.getColumnModel();
070
071        this.columnFactory = new ColumnFactory(
072                argFileAdmin,
073                tableHeader.getDefaultRenderer(),
074                this.prefs.getDateFormatter(),
075                Preferences.getTimeFormatter());
076
077        TableColumn nameCol = this.table.getColumn(
078                this.settings.getDescName());
079        int nameColNo = baseColumnModel.getColumnIndex(
080                this.settings.getDescName());
081        nameCol.setIdentifier(this.settings.getDescName());
082        nameCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
083        nameCol.setCellRenderer(
084                this.columnFactory.getLabelNameOneRenderer());
085        nameCol.setMinWidth(this.settings.getMinColumnWidth());
086        nameCol.setPreferredWidth(this.prefs.getColumnWidths(nameColNo));
087
088
089        TableColumn modifiedCol = this.table.getColumn(
090                this.settings.getDescModified());
091        int modifiedColNo = baseColumnModel.getColumnIndex(
092                this.settings.getDescModified());
093        modifiedCol.setIdentifier(this.settings.getDescModified());
094        modifiedCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
095        modifiedCol.setCellRenderer(
096                this.columnFactory.getLabelModifiedOneRenderer());
097        modifiedCol.setMinWidth(this.settings.getMinColumnWidth());
098        modifiedCol.setPreferredWidth(
099                this.prefs.getColumnWidths(modifiedColNo));
100
101
102        TableColumn importCol = this.table.getColumn(
103                this.settings.getDescImport());
104        int importColNo = baseColumnModel.getColumnIndex(
105                this.settings.getDescImport());
106        importCol.setIdentifier(this.settings.getDescImport());
107        importCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
108        importCol.setCellRenderer(this.columnFactory.getButtonOneRenderer());
109        importCol.setCellEditor(this.columnFactory.getImportOneEditor());
110        importCol.setResizable(this.settings.isButtonResizable());
111        importCol.setMinWidth(this.settings.getMinColumnWidth());
112        importCol.setPreferredWidth(this.prefs.getColumnWidths(importColNo));
113
114
115        TableColumn deleteCol = this.table.getColumn(
116                this.settings.getDescDelete());
117        int deleteColNo = baseColumnModel.getColumnIndex(
118                this.settings.getDescDelete());
119        deleteCol.setIdentifier(this.settings.getDescDelete());
120        deleteCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
121        deleteCol.setCellRenderer(this.columnFactory.getButtonOneRenderer());
122        deleteCol.setCellEditor(this.columnFactory.getDeleteOneEditor());
123        deleteCol.setResizable(this.settings.isButtonResizable());
124        deleteCol.setMinWidth(this.settings.getMinColumnWidth());
125        deleteCol.setPreferredWidth(this.prefs.getColumnWidths(deleteColNo));
126
127
128        // resizing the columns
129        TableListener tableListener = new TableListener(this.table);
130        baseColumnModel.addColumnModelListener(tableListener);
131
132        // reordering the columns
133        tableHeader.setReorderingAllowed(this.settings.isReorderingAllowed());
134
135        // sorting the columns
136        final RowSorter<TableModel> rowSorter =
137                new FileTableRowSorter(this.tableModel);
138        final List<RowSorter.SortKey> sortKeys =
139                new ArrayList<RowSorter.SortKey>();
140        sortKeys.add(this.prefs.getSortKey());
141        rowSorter.setSortKeys(sortKeys);
142        rowSorter.addRowSorterListener(tableListener);
143        this.table.setRowSorter(rowSorter);
144
145        this.table.setPreferredScrollableViewportSize(
146                new Dimension(
147                        this.settings.getMinimumTableWidth(),
148                        this.settings.getMinimumTableHeight()));
149
150        this.scrollPane = new JCustomScrollPane();
151    }
152
153    @Override
154    public JScrollPane getComponent() {
155        this.table.setRowHeight(this.prefs.getBodyRowHeight());
156        this.columnFactory.setDateFormatter(this.prefs.getDateFormatter());
157        this.columnFactory.setTimeFormatter(Preferences.getTimeFormatter());
158
159        TableColumn nameCol = this.table.getColumn(
160                this.settings.getDescName());
161        nameCol.setHeaderValue(this.localizable.getHeaderValueName());
162        TableColumn modifiedCol = this.table.getColumn(
163                this.settings.getDescModified());
164        modifiedCol.setHeaderValue(this.localizable.getHeaderValueModified());
165        TableColumn importCol = this.table.getColumn(
166                this.settings.getDescImport());
167        importCol.setHeaderValue(this.localizable.getHeaderValueImport());
168        TableColumn deleteCol = this.table.getColumn(
169                this.settings.getDescDelete());
170        deleteCol.setHeaderValue(
171                this.localizable.getHeaderValueDelete());
172
173        this.scrollPane.setViewportView(this.table);
174
175        this.scrollPane.setMinimumSize(
176                new Dimension(
177                        this.settings.getMinimumTableWidth(),
178                        this.settings.getMinimumTableHeight()));
179        this.scrollPane.setPreferredSize(
180                new Dimension(
181                        this.prefs.getPreferredTableWidth(),
182                        this.prefs.getPreferredTableHeight(
183                                this.tableModel.getRowCount())));
184        this.scrollPane.setMaximumSize(
185                new Dimension(
186                        Preferences.getMaximumTableWidth(),
187                        Preferences.getMaximumTableHeight()));
188
189        return this.scrollPane;
190    }
191
192    public ColumnFactory getColumnFactory() {
193        return this.columnFactory;
194    }
195}
