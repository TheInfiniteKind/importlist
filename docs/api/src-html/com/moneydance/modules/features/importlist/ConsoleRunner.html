001    package com.moneydance.modules.features.importlist;
002    
003    import javax.swing.JComponent;
004    import javax.swing.JFrame;
005    
006    import com.moneydance.apps.md.view.HomePageView;
007    
008    /**
009     * This class is used to run the extension as a stand-alone application from
010     * the console or from an IDE. It allows for fast feedback during the
011     * development process.
012     *
013     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
014     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
015     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
016     */
017    public final class ConsoleRunner {
018    
019        /**
020         * Private constructor prevents this class from being instantiated.
021         */
022        private ConsoleRunner() {
023        }
024    
025    
026        /**
027         * This method is called directly from the console.
028         * The first parameter may specify the base directory, but may as well
029         * be omitted.
030         * @param args First element specifies the base directory
031         */
032        public static void main(final String[] args) {
033            String baseDirectory = null;
034            if (args != null && args.length > 0) {
035                baseDirectory = args[0];
036            }
037    
038            Main main = new Main(baseDirectory);
039            HomePageView homePageView = main.getHomePageView();
040            homePageView.refresh();
041    
042            JFrame frame = new JFrame();
043            frame.setTitle(Constants.EXTENSION_NAME);
044            frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
045    
046            JComponent guiView = homePageView.getGUIView(null);
047            frame.setContentPane(guiView);
048            frame.setSize(guiView.getPreferredSize());
049            frame.setLocationRelativeTo(null);
050            frame.setVisible(true);
051        }
052    }
