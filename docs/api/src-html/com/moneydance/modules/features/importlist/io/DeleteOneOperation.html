017package com.moneydance.modules.features.importlist.io;
018
019import com.moneydance.modules.features.importlist.util.Helper;
020import com.moneydance.modules.features.importlist.util.Localizable;
021
022import java.awt.Image;
023import java.io.File;
024import java.util.List;
025import java.util.logging.Level;
026import java.util.logging.Logger;
027
028import javax.swing.Icon;
029import javax.swing.ImageIcon;
030import javax.swing.JLabel;
031import javax.swing.JOptionPane;
032
033import org.apache.commons.io.FileUtils;
034
035/**
036 * @author Florian J. Breunig
037 */
038final class DeleteOneOperation implements FileOperation {
039
040    /**
041     * Static initialization of class-dependent logger.
042     */
043    private static final Logger LOG =
044            Logger.getLogger(DeleteOneOperation.class.getName());
045
046    private final Localizable localizable;
047
048    DeleteOneOperation() {
049        this.localizable = Helper.INSTANCE.getLocalizable();
050    }
051
052    @Override
053    public void showWarningAndExecute(final List<File> files) {
054        final File file = files.iterator().next();
055        final String message =
056                this.localizable.getConfirmationMessageDeleteOneFile(
057                        file.getName());
058        final Object confirmationLabel = new JLabel(message);
059        final Image image = Helper.INSTANCE.getSettings().getIconImage();
060        Icon  icon  = null;
061        if (image != null) {
062            icon = new ImageIcon(image);
063        }
064        final Object[] options = {
065                this.localizable.getOptionDeleteFile(),
066                this.localizable.getOptionCancel()
067        };
068
069        final int choice = JOptionPane.showOptionDialog(
070                null, // no parent component
071                confirmationLabel,
072                null, // no title
073                JOptionPane.DEFAULT_OPTION,
074                JOptionPane.WARNING_MESSAGE,
075                icon,
076                options,
077                options[1]);
078
079        if (choice == 0) {
080            this.execute(files);
081        } else {
082            LOG.info(String.format(
083                    "Canceled deleting file %s",
084                    file.getAbsoluteFile()));
085        }
086    }
087
088    @Override
089    public void execute(final List<File> files) {
090        final File file = files.iterator().next();
091        // ESCA-JAVA0166: IOException, SecurityException
092        try {
093            LOG.info(String.format("Deleting file %s", file.getAbsoluteFile()));
094            FileUtils.forceDelete(file);
095        } catch (Exception e) { // $codepro.audit.disable caughtExceptions
096            LOG.log(Level.WARNING, e.getMessage(), e);
097            final String errorMessage =
098                    this.localizable.getErrorMessageDeleteFile(file.getName());
099            final Object errorLabel = new JLabel(errorMessage);
100            JOptionPane.showMessageDialog(
101                    null, // no parent component
102                    errorLabel,
103                    null, // no title
104                    JOptionPane.ERROR_MESSAGE);
105        }
106    }
107}
