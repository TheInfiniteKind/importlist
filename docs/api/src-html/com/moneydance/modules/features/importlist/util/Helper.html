017package com.moneydance.modules.features.importlist.util;
018
019import com.moneydance.apps.md.controller.FeatureModuleContext;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.util.Observable;
024import java.util.Observer;
025import java.util.logging.LogManager;
026
027import org.apache.commons.lang3.Validate;
028
029/**
030 * This singleton provides public convenience methods.
031 *
032 * @author Florian J. Breunig
033 */
034public enum Helper {
035
036    /**
037     * Helper instance.
038     */
039    INSTANCE;
040
041    private final HelperObservable observable;
042    private final Settings settings;
043    private       Preferences prefs;
044    private       Localizable localizable;
045    private       Tracker tracker;
046
047    private Helper() {
048        this.observable = new HelperObservable();
049        this.settings   = new Settings();
050    }
051
052    public Settings getSettings() {
053        return this.settings;
054    }
055
056    public Preferences getPreferences() {
057        synchronized (Helper.class) {
058            if (this.prefs == null) {
059                this.prefs = new Preferences();
060            }
061        }
062        return this.prefs;
063    }
064
065    public Localizable getLocalizable() {
066        synchronized (Helper.class) {
067            if (this.localizable == null) {
068                this.localizable = new Localizable(
069                        this.settings.getLocalizableResource(),
070                        this.prefs.getLocale());
071            }
072        }
073        return this.localizable;
074    }
075
076    public Tracker getTracker(final int build) {
077        synchronized (Helper.class) {
078            if (this.tracker == null) {
079                this.tracker = new Tracker(
080                        build,
081                        this.settings.getExtensionName(),
082                        this.prefs.getFullVersion(),
083                        this.settings.getTrackingCode());
084            }
085        }
086        return this.tracker;
087    }
088
089    public void addObserver(final Observer observer) {
090        this.observable.addObserver(observer);
091    }
092
093    public void setChanged() {
094        this.observable.setChanged();
095    }
096
097    public void notifyObservers(final Object arg) {
098        this.observable.notifyObservers(arg);
099    }
100
101    public void setContext(final FeatureModuleContext context) {
102        this.prefs.setContext(context);
103    }
104
105    public static void loadLoggerConfiguration() {
106        try {
107            InputStream inputStream = getInputStreamFromResource(
108                    Helper.INSTANCE.getSettings()
109                    .getLoggingPropertiesResource());
110            LogManager.getLogManager().readConfiguration(inputStream);
111
112        } catch (SecurityException e) {
113            e.printStackTrace(System.err);
114        } catch (IOException e) {
115            e.printStackTrace(System.err);
116        }
117    }
118
119    public static InputStream getInputStreamFromResource(
120            final String resource) {
121        ClassLoader cloader     = Helper.class.getClassLoader();
122        InputStream inputStream = cloader.getResourceAsStream(resource);
123        Validate.notNull(inputStream, "Resource %s was not found.", resource);
124        return inputStream;
125    }
126
127    /**
128     * @author Florian J. Breunig
129     */
130    private final class HelperObservable extends Observable {
131        @Override
132        public synchronized void setChanged() { // increase visiblity
133            super.setChanged();
134        }
135    }
136}
