013    public final class Helper {
014    
015        /**
016         * Private constructor prevents this class from being instantiated.
017         */
018        private Helper() {
019        }
020    
021        public static String getErrorMessageCannotReadFile(final File file) {
022            return "The file \""
023            + getTrimmedFilename(file) + "\" could not be read.";
024        }
025    
026        public static String getConfirmationMessageDeleteFile(final File file) {
027            return "Are you sure you want to delete the file \""
028            + getTrimmedFilename(file) + "\"?";
029        }
030    
031        public static String getErrorMessageCannotDeleteFile(final File file) {
032            return "The file \""
033            + getTrimmedFilename(file) + "\" could not be deleted.";
034        }
035    
036        private static String getTrimmedFilename(final File file) {
037            String filename = file.getName();
038            Preferences prefs = Preferences.getInstance();
039            int length = Math.min(filename.length(), prefs.getMaxFilenameLength());
040            String trimmedFilename = filename.substring(0, length).trim();
041            if (length < filename.length()) {
042                trimmedFilename = trimmedFilename + "É";
043            }
044            return trimmedFilename;
045        }
046    }
