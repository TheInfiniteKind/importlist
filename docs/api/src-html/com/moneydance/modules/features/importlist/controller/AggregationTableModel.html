019    package com.moneydance.modules.features.importlist.controller;
020    
021    import javax.swing.table.AbstractTableModel;
022    
023    import com.moneydance.modules.features.importlist.util.Helper;
024    import com.moneydance.modules.features.importlist.util.Localizable;
025    import com.moneydance.modules.features.importlist.util.Preferences;
026    import com.moneydance.modules.features.importlist.util.Settings;
027    
028    /**
029     * This class provides a <code>TableModel</code> implementation for a given
030     * <code>List</code> of <code>File</code>s. It takes care of the formatting and
031     * caching of its table values. The first two columns represent the name of the
032     * file and the date of its last modification, the last two columns represent
033     * the action buttons to import and delete the file, respectively.
034     *
035     * @author Florian J. Breunig
036     */
037    public final class AggregationTableModel extends AbstractTableModel {
038    
039        private static final long serialVersionUID = 3552703741263935211L;
040        private final transient Preferences prefs;
041        private final transient Settings settings;
042        private final transient Localizable localizable;
043    
044        public AggregationTableModel() {
045            this.prefs       = Helper.getPreferences();
046            this.settings    = Helper.getSettings();
047            this.localizable = Helper.getLocalizable();
048        }
049    
050        @Override
051        public Class<?> getColumnClass(final int columnIndex) {
052            return String.class;
053        }
054    
055        @Override
056        public Object getValueAt(final int row, final int column) {
057            String columnName = this.getColumnName(column);
058    
059            if (this.settings.getDescName().equals(columnName)) {
060                return null;
061            }
062            if (this.settings.getDescModified().equals(columnName)) {
063                return null;
064            }
065            if (this.settings.getDescImport().equals(columnName)) {
066                return this.localizable.getLabelImportAllButton();
067            }
068            if (this.settings.getDescDelete().equals(columnName)) {
069                return this.localizable.getLabelDeleteAllButton();
070            }
071            throw new IllegalArgumentException(
072                    "Could not find value for row " + row + ", column " + column);
073        }
074    
075        /**
076         * Only the last two columns contain buttons and are therefore editable.
077         *
078         * @param row    the row whose value is to be queried
079         * @param column the column whose value is to be queried
080         * @return true  if the column is the last or penultimate one in the model
081         */
082        @Override
083        public boolean isCellEditable(final int row, final int column) {
084            String columnName = this.getColumnName(column);
085            if (this.settings.getDescImport().equals(columnName)) {
086                return true;
087            }
088            if (this.settings.getDescDelete().equals(columnName)) {
089                return true;
090            }
091            return false;
092        }
093    
094        @Override
095        public int getColumnCount() {
096            return this.prefs.getColumnCount();
097        }
098    
099        @Override
100        public String getColumnName(final int column) {
101            return this.prefs.getColumnName(column);
102        }
103    
104        @Override
105        public int getRowCount() {
106            return 1;
107        }
108    }
