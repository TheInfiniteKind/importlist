001    package com.moneydance.modules.features.importlist;
002    
003    import java.awt.Color;
004    import java.awt.Font;
005    import java.io.File;
006    import java.text.DateFormat;
007    import java.util.Hashtable;
008    
009    import javax.swing.RowSorter;
010    import javax.swing.SortOrder;
011    import javax.swing.UIManager;
012    
013    import org.apache.commons.lang.Validate;
014    
015    import com.moneydance.apps.md.controller.Common;
016    import com.moneydance.apps.md.controller.FeatureModule;
017    import com.moneydance.apps.md.controller.FeatureModuleContext;
018    import com.moneydance.apps.md.controller.UserPreferences;
019    import com.moneydance.util.CustomDateFormat;
020    import com.moneydance.util.StreamTable;
021    
022    /**
023     * This class contains all preferences that the user might change at runtime.
024     * It serves as a facade abstracting Moneydance's
025     * <code>UserPreferences</code> (received from the
026     * <code>FeatureModuleContext</code>) and the hard-wired
027     * default values used in stand-alone mode from the rest of the extension.
028     *
029     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
030     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
031     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
032     */
033    public final class Preferences {
034    
035        private static Preferences  instance;
036        private final FeatureModule featureModule;
037        private final StreamTable   columnOrderDefault;
038        private final StreamTable   sortOrderDefault;
039        private final StreamTable   columnWidths;
040        private StreamTable         columnOrder;
041        private UserPreferences     userPreferences;
042    
043        Preferences(final FeatureModule argFeatureModule) {
044            Validate.isTrue(instance == null, "You can instantiate Preferences "
045                    + "only once");
046            Validate.notNull(argFeatureModule, "argFeatureModule can't be null");
047            this.featureModule      = argFeatureModule;
048            this.columnWidths       = new StreamTable();
049            this.columnOrderDefault = new StreamTable();
050            this.columnOrderDefault.put("0", Constants.DESC_NAME);
051            this.columnOrderDefault.put("1", Constants.DESC_MODIFIED);
052            this.columnOrderDefault.put("2", Constants.DESC_IMPORT);
053            this.columnOrderDefault.put("3", Constants.DESC_DELETE);
054            this.sortOrderDefault   = new StreamTable();
055            this.sortOrderDefault.put("0", SortOrder.ASCENDING.toString());
056            instance = this;
057        }
058    
059        public static Preferences getInstance() {
060            Validate.notNull(instance, "You must instantiate Preferences "
061                    + "first before you can access it");
062            // Reload context from feature module
063            instance.featureModule.invoke(Constants.RELOAD_CONTEXT_SUFFIX);
064            return instance;
065        }
066    
067        public void setContext(final FeatureModuleContext context) {
068            if (context != null) {
069                this.userPreferences = ((com.moneydance.apps.md.controller.Main)
070                        context).getPreferences();
071            }
072    
073            if (this.userPreferences == null) {
074                File file = Common.getPreferencesFile();
075                this.userPreferences = new UserPreferences(file);
076            }
077        }
078    
079        public void setAllWritablePreferencesToNull() {
080            this.setBaseDirectory(null);
081            this.userPreferences.setSetting(
082                    Constants.PREF_COLUMN_WIDTHS,
083                    (StreamTable) null);
084            this.userPreferences.setSetting(
085                    Constants.PREF_COLUMN_ORDER,
086                    (StreamTable) null);
087            this.userPreferences.setSetting(
088                    Constants.PREF_SORT_ORDER,
089                    (StreamTable) null);
090        }
091    
092        public String getBaseDirectory() {
093            return this.userPreferences.getSetting(
094                    Constants.PREF_BASE_DIR,
095                    null);
096        }
097    
098        public void setBaseDirectory(final String baseDirectory) {
099            this.userPreferences.setSetting(
100                    Constants.PREF_BASE_DIR,
101                    baseDirectory);
102        }
103    
104        public String getImportDirectory() {
105            return this.userPreferences.getSetting(
106                    UserPreferences.IMPORT_DIR,
107                    null);
108        }
109    
110        public void setColumnWidths(
111                final int column,
112                final int columnWidth) {
113            this.columnWidths.put(column + "", columnWidth);
114            this.userPreferences.setSetting(
115                    Constants.PREF_COLUMN_WIDTHS,
116                    this.columnWidths);
117        }
118    
119        public int getColumnWidths(final int column) {
120            StreamTable streamTable = this.userPreferences.getTableSetting(
121                    Constants.PREF_COLUMN_WIDTHS,
122                    this.columnWidths);
123            return streamTable.getInt(column + "", Constants.BUTTON_WIDTH);
124        }
125    
126        public void setColumnNames(final Hashtable<String, String> hashtable) {
127            StreamTable streamTable = new StreamTable();
128            streamTable.merge(hashtable);
129            this.userPreferences.setSetting(
130                    Constants.PREF_COLUMN_ORDER,
131                    streamTable);
132        }
133    
134        public String getColumnName(final int column) {
135            StreamTable streamTable = this.userPreferences.getTableSetting(
136                    Constants.PREF_COLUMN_ORDER,
137                    this.columnOrderDefault);
138            if (this.columnOrder == null) {
139                this.columnOrder = streamTable;
140            }
141            return this.columnOrder.getStr(column + "", null);
142        }
143    
144        public void setSortKey(final RowSorter.SortKey sortKey) {
145            StreamTable streamTable = new StreamTable();
146            streamTable.put(sortKey.getColumn(), sortKey.getSortOrder().toString());
147            this.userPreferences.setSetting(
148                    Constants.PREF_SORT_ORDER,
149                    streamTable);
150        }
151    
152        public RowSorter.SortKey getSortKey() {
153            StreamTable streamTable = this.userPreferences.getTableSetting(
154                    Constants.PREF_SORT_ORDER,
155                    null);
156            if (streamTable == null || streamTable.keySet().isEmpty()) {
157                streamTable = this.sortOrderDefault;
158            }
159            Object key        = streamTable.keys().nextElement();
160            int column        = Integer.parseInt(key + "");
161            String sortOrder  = streamTable.getStr(key, null);
162            return new RowSorter.SortKey(column, SortOrder.valueOf(sortOrder));
163        }
164    
165        public int getColumnCount() {
166            StreamTable streamTable = this.userPreferences.getTableSetting(
167                    Constants.PREF_COLUMN_ORDER,
168                    this.columnOrderDefault);
169            if (this.columnOrder == null) {
170                this.columnOrder = streamTable;
171            }
172            return this.columnOrder.size();
173        }
174    
175        public CustomDateFormat getDateFormatter() {
176            return this.userPreferences.getShortDateFormatter();
177        }
178    
179        public DateFormat getTimeFormatter() {
180            return DateFormat.getTimeInstance();
181        }
182    
183        public Color getForegroundColor() {
184            int foregroundColorValue = this.userPreferences.getIntSetting(
185                    Constants.PREF_FG_COLOR,
186                    Constants.COLOR_VALUE_FG_DEF);
187            return new Color(foregroundColorValue);
188        }
189    
190        public Color getBackgroundColor() {
191            int backgroundColorValue = this.userPreferences.getIntSetting(
192                    Constants.PREF_BG_COLOR,
193                    Constants.COLOR_VALUE_BG_DEF);
194            return new Color(backgroundColorValue);
195        }
196    
197        public Color getBackgroundColorAlt() {
198            int backgroundColorAltValue = this.userPreferences.getIntSetting(
199                    Constants.PREF_BG_COLOR_ALT,
200                    Constants.COLOR_VALUE_BG_ALT_DEF);
201            return new Color(backgroundColorAltValue);
202        }
203    
204        public Font getHeaderFont() {
205            return UIManager.getFont(Constants.HEADER_KEY_FONT);
206        }
207    
208        public Font getBodyFont() {
209            return UIManager.getFont(Constants.BODY_KEY_FONT);
210        }
211    
212        public int getHeaderRowHeight() {
213            return (int)
214            (Constants.FACTOR_ROW_HEIGHT * this.getHeaderFont().getSize());
215        }
216    
217        public int getBodyRowHeight() {
218            return (int)
219            (Constants.FACTOR_ROW_HEIGHT * this.getBodyFont().getSize());
220        }
221    }
