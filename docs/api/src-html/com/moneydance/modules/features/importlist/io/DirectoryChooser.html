001    package com.moneydance.modules.features.importlist.io;
002    import java.io.File;
003    import java.io.IOException;
004    
005    import javax.swing.JFileChooser;
006    
007    import org.apache.commons.io.FileUtils;
008    
009    import com.moneydance.modules.features.importlist.Preferences;
010    
011    
012    /**
013     * This class provides the functionality to choose, access and reset the
014     * extension's base directory. The base directory is the directory in the file
015     * system to be monitored. Choosing/resetting the base directory is reflected
016     * in the user's preferences (if there are any).
017     *
018     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
019     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
020     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
021     */
022    class DirectoryChooser {
023    
024        private static final long serialVersionUID = -8581693236906919725L;
025        private String baseDirectory;
026    
027        /**
028         * @param argBaseDirectory set the base directory when executed as a stand-
029         * alone application
030         */
031        DirectoryChooser(final String argBaseDirectory) {
032            this.baseDirectory   = argBaseDirectory;
033            if (argBaseDirectory != null) {
034                try {
035                    File baseDirectoryFile = new File(argBaseDirectory);
036                    this.baseDirectory     = baseDirectoryFile.getCanonicalPath();
037                } catch (IOException e) {
038                    e.printStackTrace(System.err);
039                }
040            }
041        }
042    
043        final void reset() {
044            this.displayFileChooser();
045            this.saveDirectoryInPreferences();
046        }
047    
048        final String getDirectory() {
049            if (this.baseDirectory == null) {
050                this.baseDirectory = Preferences.getInstance().getBaseDirectory();
051            }
052    
053            if (this.baseDirectory == null) {
054                this.displayFileChooser();
055            }
056    
057            this.saveDirectoryInPreferences();
058    
059            return this.baseDirectory;
060        }
061    
062        private void displayFileChooser() {
063            JFileChooser chooser = new JFileChooser();
064            chooser.setDialogTitle("Please choose a base directory to monitor");
065            chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
066            // disable the "All files" option.
067            chooser.setAcceptAllFileFilterUsed(false);
068    
069            chooser.setCurrentDirectory(FileUtils.getUserDirectory());
070            if (this.baseDirectory != null) {
071                File parentDirectory = new File(this.baseDirectory).getParentFile();
072                chooser.setCurrentDirectory(parentDirectory);
073            }
074    
075            if (chooser.showOpenDialog(null) != JFileChooser.APPROVE_OPTION) {
076                return;
077            }
078    
079            try {
080                this.baseDirectory = chooser.getSelectedFile().getCanonicalPath();
081            } catch (IOException e) {
082                e.printStackTrace(System.err);
083                this.baseDirectory = chooser.getSelectedFile().getAbsolutePath();
084            }
085        }
086    
087        private void saveDirectoryInPreferences() {
088            if (this.baseDirectory == null) {
089                this.baseDirectory = Preferences.getInstance().getImportDirectory();
090            }
091    
092            if (this.baseDirectory == null) {
093                this.baseDirectory = FileUtils.getUserDirectoryPath();
094            }
095    
096            Preferences.getInstance().setBaseDirectory(this.baseDirectory);
097        }
098    }
