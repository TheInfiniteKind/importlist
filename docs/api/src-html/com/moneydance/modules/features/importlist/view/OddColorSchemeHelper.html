019    package com.moneydance.modules.features.importlist.view;
020    
021    import java.awt.Color;
022    import java.awt.Component;
023    
024    import javax.swing.JComponent;
025    
026    import org.apache.commons.lang3.Validate;
027    
028    /**
029     * This helper class applies a given color scheme to a <code>Component</code>.
030     * It is used by different <code>TableCellRenderer</code>s to guarantee a
031     * consistent style in a table.
032     *
033     * @author Florian J. Breunig
034     */
035    public final class OddColorSchemeHelper implements ColorSchemeHelper {
036    
037        private Color foreground;
038        private Color background;
039        private Color backgroundAlt;
040    
041        public OddColorSchemeHelper(
042                final Color argForeground,
043                final Color argBackground,
044                final Color argBackgroundAlt) {
045            this.setForeground(argForeground);
046            this.setBackground(argBackground);
047            this.setBackgroundAlt(argBackgroundAlt);
048        }
049    
050        @Override
051        public void applyColorScheme(final Component component, final int row) {
052            component.setForeground(this.foreground);
053            component.setBackground(this.background);
054            if (row % 2 != 0) {
055                if (component instanceof JComponent) {
056                    JComponent jComponent = (JComponent) component;
057                    jComponent.setOpaque(true);
058                }
059                component.setBackground(this.backgroundAlt);
060            }
061        }
062    
063        @Override
064        public void setForeground(final Color argForeground) {
065            Validate.notNull(argForeground, "argForeground must not be null");
066            this.foreground = argForeground;
067        }
068    
069        @Override
070        public void setBackground(final Color argBackground) {
071            Validate.notNull(argBackground, "argBackground must not be null");
072            this.background = argBackground;
073        }
074    
075        @Override
076        public void setBackgroundAlt(final Color argBackgroundAlt) {
077            Validate.notNull(argBackgroundAlt, "argBackgroundAlt must not be null");
078            this.backgroundAlt = argBackgroundAlt;
079        }
080    }
