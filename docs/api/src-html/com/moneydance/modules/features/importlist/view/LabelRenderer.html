001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.awt.Component;
004    
005    import javax.swing.JLabel;
006    import javax.swing.JTable;
007    import javax.swing.table.TableCellRenderer;
008    
009    /**
010     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
011     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
012     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
013     */
014    public class LabelRenderer implements TableCellRenderer {
015    
016        private static final long serialVersionUID = 7553393304980261323L;
017    
018        public final Component getTableCellRendererComponent(
019                final JTable table,
020                final Object value,
021                final boolean isSelected,
022                final boolean hasFocus,
023                final int row,
024                final int column) {
025            JLabel label = new JLabel();
026            label.setOpaque(true);
027            ColorSchemeHelper.applyColorScheme(label, row);
028            if (value != null) {
029                label.setText(value.toString());
030            }
031            return label;
032        }
033    }
