019    package com.moneydance.modules.features.importlist.table;
020    
021    import java.awt.Component;
022    import java.awt.event.ActionEvent;
023    import java.awt.event.ActionListener;
024    
025    import javax.swing.AbstractAction;
026    import javax.swing.AbstractButton;
027    import javax.swing.Action;
028    import javax.swing.DefaultCellEditor;
029    import javax.swing.JCheckBox;
030    import javax.swing.JComponent;
031    import javax.swing.JTable;
032    import javax.swing.KeyStroke;
033    
034    import org.apache.commons.lang3.Validate;
035    
036    import com.moneydance.modules.features.importlist.io.FileAdmin;
037    import com.moneydance.modules.features.importlist.util.Helper;
038    import com.moneydance.modules.features.importlist.util.Settings;
039    
040    /**
041     * @author Florian J. Breunig
042     */
043    public abstract class AbstractEditor extends DefaultCellEditor {
044    
045        private static final long serialVersionUID = -2042274086341185241L;
046        private final transient Settings        settings;
047        private final transient FileAdmin       fileAdmin;
048        private final transient ButtonRenderer  buttonRenderer;
049        private                 String          label;
050    
051        public abstract ActionListener getActionListener(final int rowNumber);
052    
053        public abstract KeyStroke getKeyStroke();
054    
055        AbstractEditor(
056                final FileAdmin argFileAdmin,
057                final ButtonRenderer argButtonRenderer) {
058            super(new JCheckBox());
059            Validate.notNull(
060                    argFileAdmin,
061                    "argFileAdmin must not be null");
062            Validate.notNull(
063                    argButtonRenderer,
064                    "argButtonRenderer must not be null");
065            this.settings       = Helper.getSettings();
066            this.fileAdmin      = argFileAdmin;
067            this.buttonRenderer = argButtonRenderer;
068        }
069    
070        protected final Settings getSettings() {
071            return this.settings;
072        }
073    
074        protected final FileAdmin getFileAdmin() {
075            return this.fileAdmin;
076        }
077    
078        @Override
079        public final Object getCellEditorValue() {
080            return this.label;
081        }
082    
083        @Override
084        public final Component getTableCellEditorComponent(
085                final JTable table,
086                final Object value,
087                final boolean isSelected,
088                final int row,
089                final int column) {
090            if (value != null) {
091                this.label = value.toString();
092            }
093    
094            AbstractButton button = this.buttonRenderer.getTableCellRendererButton(
095                    table, value, isSelected, isSelected, row, column);
096            ActionListener actionListener = this.getActionListener(
097                    table.convertRowIndexToModel(row));
098            button.addActionListener(actionListener);
099    
100            return button;
101        }
102    
103    
104        public final void registerKeyboardShortcut(final JComponent jComponent) {
105            Validate.notNull(jComponent, "jComponent must not be null");
106            if (this.getKeyStroke() == null) {
107                return;
108            }
109    
110            final Action action = new AbstractAction() {
111                private static final long serialVersionUID = 947999658820121305L;
112    
113                @Override
114                public void actionPerformed(final ActionEvent actionEvent) {
115                    ActionListener actionListener =
116                        AbstractEditor.this.getActionListener(0);
117                    actionListener.actionPerformed(actionEvent);
118                }
119            };
120    
121            final String actionMapKey = this.getClass().getName(); // unique
122            jComponent.getInputMap(JComponent.WHEN_IN_FOCUSED_WINDOW).put(
123                    this.getKeyStroke(),
124                    actionMapKey);
125            jComponent.getActionMap().put(actionMapKey, action);
126        }
127    }
