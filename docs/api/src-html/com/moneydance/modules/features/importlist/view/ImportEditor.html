001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.awt.Component;
004    import java.awt.event.ActionListener;
005    
006    import javax.swing.AbstractButton;
007    import javax.swing.DefaultCellEditor;
008    import javax.swing.JCheckBox;
009    import javax.swing.JTable;
010    
011    import com.moneydance.modules.features.importlist.io.FileAdministration;
012    
013    /**
014     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
015     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
016     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
017     */
018    public class ImportEditor extends DefaultCellEditor {
019    
020        private static final long serialVersionUID = 4286061728240721119L;
021        private final FileAdministration    fileAdministration;
022        private final ButtonRenderer        buttonRenderer;
023        private String label;
024    
025        ImportEditor(final FileAdministration argFileAdministration,
026                final ButtonRenderer argButtonRenderer) {
027            super(new JCheckBox());
028            this.fileAdministration = argFileAdministration;
029            this.buttonRenderer     = argButtonRenderer;
030        }
031    
032        @Override
033        public final Object getCellEditorValue() {
034            return this.label;
035        }
036    
037        @Override
038        public final Component getTableCellEditorComponent(
039                final JTable table,
040                final Object value,
041                final boolean isSelected,
042                final int row,
043                final int column) {
044            if (value != null) {
045                this.label = value.toString();
046            }
047    
048            AbstractButton button = this.buttonRenderer.getTableCellRendererButton(
049                    table, value, isSelected, isSelected, row, column);
050            ActionListener actionListener =
051                this.fileAdministration.getImportActionListener(
052                        table.convertRowIndexToModel(row));
053            button.addActionListener(actionListener);
054    
055            return button;
056        }
057    }
