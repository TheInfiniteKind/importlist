017package com.moneydance.modules.features.importlist.io;
018
019import com.infinitekind.moneydance.model.Account;
020import com.infinitekind.moneydance.model.AccountUtil;
021import com.infinitekind.moneydance.model.AcctFilter;
022import com.moneydance.apps.md.controller.FeatureModuleContext;
023import com.moneydance.modules.features.importlist.util.Helper;
024import com.moneydance.modules.features.importlist.util.Preferences;
025
026import java.io.File;
027import java.io.FileFilter;
028import java.util.HashMap;
029import java.util.Iterator;
030import java.util.List;
031import java.util.Map;
032import java.util.logging.Logger;
033
034import org.apache.commons.io.FilenameUtils;
035import org.apache.commons.lang3.text.StrSubstitutor;
036
037/**
038 * @author Florian J. Breunig
039 */
040final class ImportOneOperation implements FileOperation {
041
042    private final Preferences          prefs;
043    private final FeatureModuleContext context;
044    private final FileFilter           transactionFileFilter;
045    private final FileFilter           textFileFilter;
046
047    /**
048     * Static initialization of class-dependent logger.
049     */
050    private static final Logger LOG =
051            Logger.getLogger(ImportOneOperation.class.getName());
052
053    ImportOneOperation(
054            final FeatureModuleContext argContext,
055            final FileFilter argTransactionFileFilter,
056            final FileFilter argTextFileFilter) {
057        this.prefs                  = Helper.INSTANCE.getPreferences();
058        this.context                = argContext;
059        this.transactionFileFilter  = argTransactionFileFilter;
060        this.textFileFilter         = argTextFileFilter;
061    }
062
063    @Override
064    public void showWarningAndExecute(final List<File> files) {
065        this.execute(files);
066    }
067
068    @Override
069    public void execute(final List<File> files) {
070        final File file = files.iterator().next();
071        Map<String, String> valuesMap = new HashMap<String, String>(1);
072        valuesMap.put("filename",  file.getAbsolutePath());
073
074        String uriScheme = "";
075        if (this.transactionFileFilter.accept(file)) {
076            uriScheme = Helper.INSTANCE.getSettings()
077                    .getTransactionFileImportUriScheme();
078        } else if (this.textFileFilter.accept(file)) {
079            uriScheme = Helper.INSTANCE.getSettings()
080                    .getTextFileImportUriScheme();
081            valuesMap.put("accountno", this.getAccountNumberForFile(file));
082        }
083
084        final StrSubstitutor sub = new StrSubstitutor(valuesMap);
085        final String resolvedUri = sub.replace(uriScheme);
086
087        // Import the file by calling the URI
088        this.context.showURL(resolvedUri);
089    }
090
091    private String getAccountNumberForFile(final File argFile) {
092        final String fileName = FilenameUtils.removeExtension(
093                argFile.getName());
094
095        Account rootAccount =
096                this.context.getCurrentAccountBook().getRootAccount();
097
098        String accountNumber = "-1";
099
100        for (Iterator<Account> iterator =
101                AccountUtil.getAccountIterator(rootAccount);
102                iterator.hasNext();) {
103            final Account account = iterator.next();
104            if (!AcctFilter.NON_CATEGORY_FILTER.matches(account)
105                    || rootAccount.equals(account)) {
106                continue;
107            }
108
109            final String accountName = account.getFullAccountName();
110            if (this.isEqual(accountName, fileName)) {
111                LOG.config(String.format(
112                        "Found matching account \"%s\" for file %s",
113                        account.getFullAccountName(),
114                        argFile.getName()));
115                accountNumber = String.valueOf(account.getAccountNum());
116            }
117        }
118
119        return accountNumber;
120    }
121
122    private boolean isEqual(final String first, final String second) {
123        final String string1 =  first.toUpperCase(this.prefs.getLocale());
124        final String string2 = second.toUpperCase(this.prefs.getLocale());
125        return string1.contains(string2) || string2.contains(string1);
126    }
127}
