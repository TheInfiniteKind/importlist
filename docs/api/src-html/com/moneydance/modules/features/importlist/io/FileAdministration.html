031    /**
032     * This core class coordinates and delegates operations on the file system.
033     *
034     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
035     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
036     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
037     */
038    public class FileAdministration {
039    
040        /**
041         * Static initialization of class-dependent logger.
042         */
043        private static Logger log = Logger.getLogger(FileAdministration.class);
044    
045        private final Preferences               prefs;
046        private final FeatureModule             featureModule;
047        private FeatureModuleContext            context;
048        private final DirectoryChooser          directoryChooser;
049        private final IOFileFilter              fileFilter;
050        private FileAlterationObserver          observer;
051        private final TransactionFileListener   listener;
052        private final FileAlterationMonitor     monitor;
053        private final List<File>                files;
054        private boolean                         dirty;
055        private HomePageView                    homePageView;
056    
057        public FileAdministration(final FeatureModule argFeatureModule,
058                final String baseDirectory) {
059            Validate.notNull(argFeatureModule, "argFeatureModule can't be null");
060            this.prefs            = Preferences.getInstance();
061            this.featureModule    = argFeatureModule;
062            this.directoryChooser = new DirectoryChooser(baseDirectory);
063            this.fileFilter       = new AndFileFilter(
064                    CanReadFileFilter.CAN_READ,
065                    new SuffixFileFilter(
066                            this.prefs.getFileExtensions(),
067                            IOCase.INSENSITIVE)
068            );
069    
070            this.listener = new TransactionFileListener(this);
071            this.monitor  = new FileAlterationMonitor(
072                    this.prefs.getMonitorIntervall());
073            this.setFileMonitorToCurrentImportDir();
074            this.startMonitor();
075    
076            this.files = new ArrayList<File>();
077    
078            this.dirty = true;
079        }
080    
081        public final void setContext(final FeatureModuleContext argContext) {
082            this.context = argContext;
083        }
084    
085        public final void reset() {
086            log.info("Resetting base directory.");
087            this.directoryChooser.reset();
088            this.monitor.removeObserver(this.observer);
089            this.setFileMonitorToCurrentImportDir();
090            this.setDirty(true);
091        }
092    
093        public final List<File> getFiles() {
094            return this.files;
095        }
096    
097        public final void reloadFiles() {
098            this.files.clear();
099            try {
100                File importDir = new File(this.getImportDir());
101                Collection<File> collection = FileUtils.listFiles(
102                        importDir,
103                        this.fileFilter,
104                        null); // ignore subdirectories
105                this.files.addAll(collection);
106            } catch (IllegalArgumentException e) {
107                log.warn(e.getMessage(), e);
108            }
109        }
110    
111        public final String getImportDir() {
112            return this.directoryChooser.getDirectory();
113        }
114    
115        public final void setDirty(final boolean argDirty) {
116            this.dirty = argDirty;
117            if (argDirty && this.homePageView != null) {
118                this.reloadFiles();
119                this.homePageView.refresh();
120            }
121        }
122    
123        public final boolean isDirty() {
124            return this.dirty;
125        }
126    
127        /**
128         * Start monitoring the current import directory.
129         */
130        public final void startMonitor() {
131            log.debug("Starting the directory monitor.");
132            try {
133                this.monitor.start();
134            } catch (Exception e) {
135                log.warn(e.getMessage(), e);
136            }
137        }
138    
139        /**
140         * Stop monitoring the current import directory.
141         */
142        public final void stopMonitor() {
143            log.debug("Stopping the directory monitor.");
144            try {
145                this.monitor.stop();
146            } catch (Exception e) {
147                log.warn(e.getMessage(), e);
148            }
149        }
150    
151        public final void setHomePageView(final HomePageView argHomePageView) {
152            this.homePageView = argHomePageView;
153        }
154    
155        public final ActionListener getImportActionListener(final int rowNumber) {
156            return this.new ImportActionListener(rowNumber);
157        }
158    
159        public final ActionListener getDeleteActionListener(final int rowNumber) {
160            return this.new DeleteActionListener(rowNumber);
161        }
162    
163        private void setFileMonitorToCurrentImportDir() {
164            this.observer  = new FileAlterationObserver(
165                    this.getImportDir(),
166                    this.fileFilter,
167                    IOCase.SENSITIVE);
168            this.observer.addListener(this.listener);
169            this.monitor.addObserver(this.observer);
170        }
171    
172        /**
173         * Command pattern: Return an action that imports a file identified by its
174         * position in the list.
175         */
176        private final class ImportActionListener implements ActionListener {
177    
178            private final int rowNumber;
179    
180            private ImportActionListener(final int argRowNumber) {
181                this.rowNumber = argRowNumber;
182            }
183    
184            @Override
185            public void actionPerformed(final ActionEvent actionEvent) {
186                File file = FileAdministration.this.getFiles().get(this.rowNumber);
187                log.info("Importing file " + file.getAbsoluteFile());
188    
189                if (!file.canRead()) {
190                    final String errorMessage =
191                        FileAdministration.this.prefs.getErrorMessageCannotReadFile(
192                                file);
193                    log.warn(errorMessage);
194                    JOptionPane.showMessageDialog(
195                            null, // no parent component
196                            errorMessage,
197                            null, // no title
198                            JOptionPane.ERROR_MESSAGE);
199                    return;
200                }
201    
202                if (FileAdministration.this.context == null) {
203                    log.warn("Could not import file " + file
204                            + " since context is not set.");
205                    return;
206                }
207    
208                String callUri = FileAdministration.this.prefs.getImportUriPrefix()
209                + file.getAbsolutePath();
210    
211                // Import the file identified by the file parameter
212                FileAdministration.this.context.showURL(callUri);
213            }
214        }
215    
216        /**
217         * Command pattern: Return an action that deletes a file identified by its
218         * position in the list.
219         */
220        private final class DeleteActionListener implements ActionListener {
221    
222            private final int rowNumber;
223    
224            private DeleteActionListener(final int argRowNumber) {
225                this.rowNumber = argRowNumber;
226            }
227    
228            @Override
229            public void actionPerformed(final ActionEvent actionEvent) {
230                File file = FileAdministration.this.getFiles().get(this.rowNumber);
231    
232                final String confirmationMessage =
233                    FileAdministration.this.prefs.getConfirmationMessageDeleteFile(
234                            file);
235                Icon icon   = null;
236                Image image = FileAdministration.this.featureModule.getIconImage();
237                if (image != null) {
238                    icon = new ImageIcon(image);
239                }
240                Object[] options = {
241                        FileAdministration.this.prefs.getOptionDeleteFile(),
242                        FileAdministration.this.prefs.getOptionCancel()
243                };
244    
245                int choice = JOptionPane.showOptionDialog(
246                        null, // no parent component
247                        confirmationMessage,
248                        null, // no title
249                        JOptionPane.DEFAULT_OPTION,
250                        JOptionPane.WARNING_MESSAGE,
251                        icon,
252                        options,
253                        options[1]);
254    
255                if (choice == 0) {
256                    log.info("Deleting file " + file.getAbsoluteFile());
257    
258                    if (file.delete()) {
259                        log.info("Deleted file " + file.getAbsoluteFile());
260                    } else {
261                        final String errorMessage =
262                            FileAdministration.this.prefs.getErrorMessageDeleteFile(
263                                    file);
264                        log.warn(errorMessage);
265                        JOptionPane.showMessageDialog(
266                                null, // no parent component
267                                errorMessage,
268                                null, // no title
269                                JOptionPane.ERROR_MESSAGE);
270                    }
271                } else {
272                    log.info("Canceled deleting file " + file.getAbsoluteFile());
273                }
274    
275                FileAdministration.this.setDirty(true);
276            }
277        }
