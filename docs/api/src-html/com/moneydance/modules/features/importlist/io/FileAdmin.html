019    package com.moneydance.modules.features.importlist.io;
020    
021    import java.awt.Image;
022    import java.io.File;
023    import java.io.IOException;
024    import java.util.ArrayList;
025    import java.util.Collection;
026    import java.util.Collections;

028    import java.util.Observable;

031    import javax.swing.Icon;

036    import org.apache.commons.io.FileUtils;
037    import org.apache.commons.io.IOCase;
038    import org.apache.commons.io.filefilter.AbstractFileFilter;
039    import org.apache.commons.io.filefilter.AndFileFilter;
040    import org.apache.commons.io.filefilter.CanReadFileFilter;
041    import org.apache.commons.io.filefilter.OrFileFilter;
042    import org.apache.commons.io.filefilter.SuffixFileFilter;

047    import com.moneydance.apps.md.controller.FeatureModuleContext;

049    import com.moneydance.modules.features.importlist.util.Preferences;

051    /**
052     * This core class coordinates and delegates operations on the file system.
053     */
054    public final class FileAdmin extends Observable implements Observer {
055    
056        /**
057         * Static initialization of class-dependent logger.
058         */
059        private static final Logger LOG = Logger.getLogger(FileAdmin.class);
060    
061        private final Preferences               prefs;
062        private final FeatureModuleContext      context;
063        private final DirectoryChooser          directoryChooser;
064        private final AbstractFileFilter        transactionFileFilter;
065        private final AbstractFileFilter        textFileFilter;
066        private final AbstractFileFilter        readableFileFilter;
067        private FileAlterationObserver          observer;
068        private final TransactionFileListener   listener;
069        private final FileAlterationMonitor     monitor;
070        private final List<File>                files;
071    
072        public FileAdmin(final String baseDirectory,
073                final FeatureModuleContext argContext) {
074            super();
075            this.prefs            = Helper.getPreferences();
076            this.context          = argContext;
077            this.directoryChooser = new DirectoryChooser(baseDirectory);
078            this.transactionFileFilter = new SuffixFileFilter(
079                    this.prefs.getTransactionFileExtensions(),
080                    IOCase.INSENSITIVE);
081            this.textFileFilter = new SuffixFileFilter(
082                    this.prefs.getTextFileExtensions(),
083                    IOCase.INSENSITIVE);
084            this.readableFileFilter = new AndFileFilter(
085                    CanReadFileFilter.CAN_READ,
086                    new OrFileFilter(
087                            this.transactionFileFilter,
088                            this.textFileFilter));
089    
090            this.listener = new TransactionFileListener();
091            this.listener.addObserver(this);
092            this.monitor  = new FileAlterationMonitor(
093                    this.prefs.getMonitorIntervall());
094    
095            this.files = Collections.synchronizedList(new ArrayList<File>());
096        }
097    
098        public void resetBaseDirectory() {
099            LOG.info("Resetting base directory.");
100            this.directoryChooser.reset();
101            this.monitor.removeObserver(this.observer);
102            this.setFileMonitorToCurrentImportDir();
103            this.setChanged();
104            this.notifyObservers();
105        }
106    
107        public List<File> getFiles() {
108            return Collections.unmodifiableList(this.files);
109        }
110    
111        public synchronized void reloadFiles() {
112            this.files.clear();
113            try {
114                Collection<File> collection = FileUtils.listFiles(
115                        this.getBaseDirectory(),
116                        this.readableFileFilter,
117                        null); // ignore subdirectories
118                this.files.addAll(collection);
119            } catch (IllegalArgumentException e) {
120                LOG.warn(e.getMessage(), e);
121            }
122        }
123    
124        public File getBaseDirectory() {
125            return this.directoryChooser.getBaseDirectory();
126        }
127    
128        @Override
129        public void update(final Observable observable, final Object object) {
130            this.setChanged();
131            this.notifyObservers();
132        }
133    
134        /**
135         * Start monitoring the current import directory.
136         */
137        public void startMonitor() {
138            LOG.debug("Starting the directory monitor.");
139            this.setFileMonitorToCurrentImportDir();
140            try {
141                this.monitor.start();
142            } catch (Exception e) {
143                LOG.warn(e.getMessage(), e);
144            }
145        }
146    
147        /**
148         * Stop monitoring the current import directory.
149         */
150        public void stopMonitor() {
151            LOG.debug("Stopping the directory monitor.");
152            try {
153                this.monitor.stop();
154            } catch (Exception e) {
155                LOG.warn(e.getMessage(), e);
156            }
157        }
158    
159        public void importRow(final int rowNumber) {
160            if (rowNumber >= this.files.size()) {
161                return;
162            }
163            File file = this.files.get(rowNumber);
164            LOG.info("Importing file " + file.getAbsoluteFile());
165    
166            String callUri = "";
167            if (this.transactionFileFilter.accept(file)) {
168                callUri = this.prefs.getTransactionFileImportUriPrefix()
169                + file.getAbsolutePath();
170            }
171            if (this.textFileFilter.accept(file)) {
172                callUri = this.prefs.getTextFileImportUriPrefix()
173                + file.getAbsolutePath();
174            }
175            // Import the file identified by the file parameter
176            this.context.showURL(callUri);
177    
178            this.setChanged();
179            this.notifyObservers();
180        }
181    
182        public void deleteRow(final int rowNumber) {
183            if (rowNumber >= this.files.size()) {
184                return;
185            }
186            final File file = this.files.get(rowNumber);
187    
188            final String confirmationMessage =
189                this.prefs.getConfirmationMessageDeleteFile(file.getName());
190            Object confirmationLabel = new JLabel(confirmationMessage);
191            Icon  icon  = null;
192            Image image = Helper.getIconImage();
193            if (image != null) {
194                icon = new ImageIcon(image);
195            }
196            Object[] options = {
197                    this.prefs.getOptionDeleteFile(),
198                    this.prefs.getOptionCancel()
199            };
200    
201            int choice = JOptionPane.showOptionDialog(
202                    null, // no parent component
203                    confirmationLabel,
204                    null, // no title
205                    JOptionPane.DEFAULT_OPTION,
206                    JOptionPane.WARNING_MESSAGE,
207                    icon,
208                    options,
209                    options[1]);
210    
211            try {
212                if (choice == 0) {
213                    FileUtils.forceDelete(file);
214                    LOG.info("Deleted file " + file.getAbsoluteFile());
215                } else {
216                    LOG.info("Canceled deleting file " + file.getAbsoluteFile());
217                }
218            } catch (IOException e) {
219                LOG.warn("Could not delete file " + file.getAbsoluteFile());
220                final String errorMessage =
221                    this.prefs.getErrorMessageDeleteFile(file.getName());
222                Object errorLabel = new JLabel(errorMessage);
223                JOptionPane.showMessageDialog(
224                        null, // no parent component
225                        errorLabel,
226                        null, // no title
227                        JOptionPane.ERROR_MESSAGE);
228            }
229            this.setChanged();
230            this.notifyObservers();
231        }
232    
233        private void setFileMonitorToCurrentImportDir() {
234            this.observer  = new FileAlterationObserver(
235                    this.getBaseDirectory(),
236                    this.readableFileFilter,
237                    IOCase.SYSTEM);
238            this.observer.addListener(this.listener);
239            this.monitor.addObserver(this.observer);
240        }
241    }
