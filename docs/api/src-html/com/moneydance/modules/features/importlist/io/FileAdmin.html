021    import java.awt.Image;
022    import java.awt.event.ActionEvent;
023    import java.awt.event.ActionListener;

025    import java.util.ArrayList;
026    import java.util.Collection;
027    import java.util.List;
028    
029    import javax.swing.Icon;
030    import javax.swing.ImageIcon;

034    import org.apache.commons.io.FileUtils;
035    import org.apache.commons.io.IOCase;
036    import org.apache.commons.io.filefilter.AbstractFileFilter;
037    import org.apache.commons.io.filefilter.AndFileFilter;
038    import org.apache.commons.io.filefilter.CanReadFileFilter;
039    import org.apache.commons.io.filefilter.SuffixFileFilter;
040    import org.apache.commons.io.monitor.FileAlterationMonitor;
041    import org.apache.commons.io.monitor.FileAlterationObserver;

045    import com.moneydance.apps.md.controller.FeatureModule;

048    import com.moneydance.modules.features.importlist.util.Preferences;
049    
050    /**
051     * This core class coordinates and delegates operations on the file system.
052     */
053    public class FileAdmin {
054    
055        /**
056         * Static initialization of class-dependent logger.
057         */
058        private static Logger log = Logger.getLogger(FileAdmin.class);
059    
060        private final Preferences               prefs;
061        private final FeatureModule             featureModule;
062        private FeatureModuleContext            context;
063        private final DirectoryChooser          directoryChooser;
064        private final AbstractFileFilter        fileFilter;
065        private FileAlterationObserver          observer;
066        private final TransactionFileListener   listener;
067        private final FileAlterationMonitor     monitor;
068        private final List<File>                files;
069        private boolean                         dirty;
070        private HomePageView                    homePageView;
071    
072        public FileAdmin(final FeatureModule argFeatureModule,
073                final String baseDirectory) {
074            Validate.notNull(argFeatureModule, "argFeatureModule can't be null");
075            this.prefs            = Preferences.getInstance();
076            this.featureModule    = argFeatureModule;
077            this.directoryChooser = new DirectoryChooser(baseDirectory);
078            this.fileFilter       = new AndFileFilter(
079                    CanReadFileFilter.CAN_READ,
080                    new SuffixFileFilter(
081                            this.prefs.getFileExtensions(),
082                            IOCase.INSENSITIVE)
083            );
084    
085            this.listener = new TransactionFileListener(this);
086            this.monitor  = new FileAlterationMonitor(
087                    this.prefs.getMonitorIntervall());
088            this.setFileMonitorToCurrentImportDir();
089            this.startMonitor();
090    
091            this.files = new ArrayList<File>();
092    
093            this.dirty = true;
094        }
095    
096        public final void setContext(final FeatureModuleContext argContext) {
097            this.context = argContext;
098        }
099    
100        public final void reset() {
101            log.info("Resetting base directory.");
102            this.directoryChooser.reset();
103            this.monitor.removeObserver(this.observer);
104            this.setFileMonitorToCurrentImportDir();
105            this.setDirty(true);
106        }
107    
108        public final void deleteFile(final File file) {
109            final String confirmationMessage =
110                this.prefs.getConfirmationMessageDeleteFile(file);
111            Object confirmationLabel = new JLabel(confirmationMessage);
112            Icon icon   = null;
113            Image image = this.featureModule.getIconImage();
114            if (image != null) {
115                icon = new ImageIcon(image);
116            }
117            Object[] options = {
118                    this.prefs.getOptionDeleteFile(),
119                    this.prefs.getOptionCancel()
120            };
121    
122            int choice = JOptionPane.showOptionDialog(
123                    null, // no parent component
124                    confirmationLabel,
125                    null, // no title
126                    JOptionPane.DEFAULT_OPTION,
127                    JOptionPane.WARNING_MESSAGE,
128                    icon,
129                    options,
130                    options[1]);
131    
132            if (choice == 0) {
133                log.info("Deleting file " + file.getAbsoluteFile());
134    
135                if (file.delete()) {
136                    log.info("Deleted file " + file.getAbsoluteFile());
137                } else {
138                    log.warn("Could not delete file "
139                            + file.getAbsoluteFile());
140                    final String errorMessage =
141                        this.prefs.getErrorMessageDeleteFile(file);
142                    Object errorLabel = new JLabel(errorMessage);
143                    JOptionPane.showMessageDialog(
144                            null, // no parent component
145                            errorLabel,
146                            null, // no title
147                            JOptionPane.ERROR_MESSAGE);
148                }
149            } else {
150                log.info("Canceled deleting file " + file.getAbsoluteFile());
151            }
152    
153            this.setDirty(true);
154        }
155    
156        public final List<File> getFiles() {
157            return this.files;
158        }
159    
160        public final void reloadFiles() {
161            this.files.clear();
162            try {
163                Collection<File> collection = FileUtils.listFiles(
164                        this.getBaseDirectory(),
165                        this.fileFilter,
166                        null); // ignore subdirectories
167                this.files.addAll(collection);
168            } catch (IllegalArgumentException e) {
169                log.warn(e.getMessage(), e);
170            }
171        }
172    
173        public final File getBaseDirectory() {
174            return this.directoryChooser.getBaseDirectory();
175        }
176    
177        public final void setDirty(final boolean argDirty) {
178            this.dirty = argDirty;
179            if (this.dirty && this.homePageView != null) {
180                log.info("Base directory contains changes.");
181                this.reloadFiles();
182                this.homePageView.refresh();
183            }
184        }
185    
186        public final boolean isDirty() {
187            return this.dirty;
188        }
189    
190        /**
191         * Start monitoring the current import directory.
192         */
193        public final void startMonitor() {
194            log.debug("Starting the directory monitor.");
195            try {
196                this.monitor.start();
197            } catch (Exception e) {
198                log.warn(e.getMessage(), e);
199            }
200        }
201    
202        /**
203         * Stop monitoring the current import directory.
204         */
205        public final void stopMonitor() {
206            log.debug("Stopping the directory monitor.");
207            try {
208                this.monitor.stop();
209            } catch (Exception e) {
210                log.warn(e.getMessage(), e);
211            }
212        }
213    
214        public final void setHomePageView(final HomePageView argHomePageView) {
215            this.homePageView = argHomePageView;
216        }
217    
218        public final ActionListener getImportActionListener(final int rowNumber) {
219    
220            return new ActionListener() {
221                @Override
222                public void actionPerformed(final ActionEvent actionEvent) {
223                    File file = FileAdmin.this.files.get(rowNumber);
224                    log.info("Importing file " + file.getAbsoluteFile());
225    
226                    String callUri = FileAdmin.this.prefs.getImportUriPrefix()
227                    + file.getAbsolutePath();
228    
229                    // Import the file identified by the file parameter
230                    FileAdmin.this.context.showURL(callUri);
231                }
232            };
233        }
234    
235        public final ActionListener getDeleteActionListener(final int rowNumber) {
236    
237            return new ActionListener() {
238                @Override
239                public void actionPerformed(final ActionEvent actionEvent) {
240                    File file = FileAdmin.this.files.get(rowNumber);
241                    log.info("Deleting file " + file.getAbsoluteFile());
242    
243                    String callUri = FileAdmin.this.prefs.getId() + ":"
244                    + FileAdmin.this.prefs.getDeleteFileSuffix() + "?"
245                    + file.getAbsolutePath();
246    
247                    // Delete the file identified by the file parameter
248                    FileAdmin.this.context.showURL(callUri);
249                }
250            };
251        };
252    
253        private void setFileMonitorToCurrentImportDir() {
254            this.observer  = new FileAlterationObserver(
255                    this.getBaseDirectory(),
256                    this.fileFilter,
257                    IOCase.SENSITIVE);
258            this.observer.addListener(this.listener);
259            this.monitor.addObserver(this.observer);
260        }
261    }
