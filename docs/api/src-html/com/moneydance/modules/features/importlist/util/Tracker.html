019    package com.moneydance.modules.features.importlist.util;
020    
021    import java.net.Authenticator;
022    import java.net.InetSocketAddress;
023    import java.net.PasswordAuthentication;
024    import java.net.Proxy;
025    import java.net.SocketAddress;
026    
027    import org.slf4j.Logger;
028    import org.slf4j.LoggerFactory;
029    
030    import com.dmurph.tracking.AnalyticsConfigData;
031    import com.dmurph.tracking.JGoogleAnalyticsTracker;
032    import com.dmurph.tracking.JGoogleAnalyticsTracker.GoogleAnalyticsVersion;
033    
034    /**
035     * @author Florian J. Breunig
036     */
037    public final class Tracker {
038    
039        private final Preferences prefs;
040        private final String fullVersion;
041        private final String build;
042    
043        private final JGoogleAnalyticsTracker analyticsTracker;
044        private static final Logger LOG = LoggerFactory.getLogger(Tracker.class);
045    
046        Tracker(final int argBuild) {
047            this.prefs          = Helper.INSTANCE.getPreferences();
048            this.fullVersion    = "Moneydance " + this.prefs.getFullVersion();
049            this.build          = "Import List v" + String.valueOf(argBuild);
050    
051            JGoogleAnalyticsTracker.setProxy(this.getProxy());
052            Settings settings = Helper.INSTANCE.getSettings();
053            AnalyticsConfigData config = new AnalyticsConfigData(
054                    settings.getTrackingCode());
055            this.analyticsTracker = new JGoogleAnalyticsTracker(
056                    config,
057                    GoogleAnalyticsVersion.V_4_7_2);
058        }
059    
060        public void track(final EventName eventName) {
061            LOG.debug("trackEvent");
062            this.analyticsTracker.trackEvent(
063                    this.fullVersion,
064                    eventName.toString(),
065                    this.build);
066        }
067    
068        private Proxy getProxy() {
069            if (!this.prefs.useProxy()) {
070                return Proxy.NO_PROXY;
071            }
072    
073            final SocketAddress socketAddress = new InetSocketAddress(
074                    this.prefs.getProxyHost(),
075                    this.prefs.getProxyPort());
076    
077            boolean authProxy = this.prefs.needProxyAuthentication();
078    
079            Proxy.Type proxyType = Proxy.Type.HTTP;
080    
081            if (authProxy) {
082                proxyType = Proxy.Type.SOCKS;
083                Authenticator.setDefault(new Authenticator() {
084                    @Override
085                    protected PasswordAuthentication getPasswordAuthentication() {
086                        return new PasswordAuthentication(
087                               Tracker.this.prefs.getProxyUsername(),
088                               Tracker.this.prefs.getProxyPassword().toCharArray());
089                    }
090                });
091            }
092            return new Proxy(proxyType, socketAddress);
093        }
094    
095        /**
096         * @author Florian J. Breunig
097         */
098        public enum EventName {
099    
100            INSTALL(Helper.INSTANCE.getSettings().getEventActionInstall()),
101    
102            DISPLAY(Helper.INSTANCE.getSettings().getEventActionDisplay()),
103    
104            UNINSTALL(Helper.INSTANCE.getSettings().getEventActionUninstall());
105    
106            private final String eventString;
107    
108            private EventName(final String argEventString) {
109                this.eventString = argEventString;
110            }
111    
112            @Override
113            public String toString() {
114                return this.eventString;
115            }
116        }
117    }
