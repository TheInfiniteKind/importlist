079        public ViewController(
080                final String baseDirectory,
081                final FeatureModuleContext context,
082                final int build) {
083            this.prefs       = Helper.getPreferences();
084            this.settings    = Helper.getSettings();
085            this.localizable = Helper.getLocalizable();
086            this.fileAdmin   = new FileAdmin(baseDirectory, context);
087            this.fileAdmin.addObserver(this);
088            this.tracker     = new Tracker(
089                    this.prefs.getFullVersion(),
090                    build,
091                    this.settings.getTrackingCode());
092        }

095        private void init() {
096            this.viewport = new JViewport();
097            this.viewport.setOpaque(false);
098    
099            this.emptyLabel = new JLabel();
100            this.emptyLabel.setHorizontalAlignment(SwingConstants.CENTER);
101            this.emptyLabel.setBorder(MoneydanceLAF.homePageBorder);
102    
103            this.splitPane = new JCustomSplitPane(JSplitPane.VERTICAL_SPLIT);
104            this.splitPane.setResizeWeight(1.0);

109    
110            this.baseTableModel = new FileTableModel(this.fileAdmin.getFiles());
111

113            this.baseTable.setOpaque(false);

122            this.aggrTableModel = new AggregationTableModel();
123            this.aggrTable = new JTable(this.aggrTableModel);
124            this.aggrTable.setOpaque(false);
125            this.aggrTable.setShowGrid(false);

128            this.aggrTable.setIntercellSpacing(
129                    new Dimension(

164                    this.columnFactory.getLabelNameAllRenderer());
165            nameCol.setMinWidth(this.settings.getMinColumnWidth());
166            nameCol.setPreferredWidth(this.prefs.getColumnWidths(nameColNo));
167    
168    
169            TableColumn modifiedCol = this.baseTable.getColumn(
170                    this.settings.getDescModified());
171            int modifiedColNo = baseColumnModel.getColumnIndex(
172                    this.settings.getDescModified());
173            modifiedCol.setIdentifier(this.settings.getDescModified());
174            modifiedCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
175            modifiedCol.setCellRenderer(
176                    this.columnFactory.getLabelModifiedOneRenderer());
177            modifiedCol.setMinWidth(this.settings.getMinColumnWidth());
178            modifiedCol.setPreferredWidth(
179                    this.prefs.getColumnWidths(modifiedColNo));

182                    this.settings.getDescModified());

273        @Override
274        public synchronized void refresh() {
275            if (this.scrollPane == null || !this.scrollPane.isVisible()) {
276                return;
277            }
278    
279            this.prefs.reload();
280            this.columnFactory.setDateFormatter(this.prefs.getDateFormatter());
281            this.columnFactory.setTimeFormatter(this.prefs.getTimeFormatter());
282            this.columnFactory.setForeground(this.prefs.getForeground());
283            this.columnFactory.setBackground(this.prefs.getBackground());
284            this.columnFactory.setBackgroundAlt(this.prefs.getBackgroundAlt());
285            this.scrollPane.setBackground(this.prefs.getBackground());
286            TableColumn nameCol = this.baseTable.getColumn(
287                    this.settings.getDescName());
288            nameCol.setHeaderValue(this.localizable.getHeaderValueName());
289            TableColumn modifiedCol = this.baseTable.getColumn(
290                    this.settings.getDescModified());
291            modifiedCol.setHeaderValue(this.localizable.getHeaderValueModified());
292            TableColumn importCol = this.baseTable.getColumn(
293                    this.settings.getDescImport());
294            importCol.setHeaderValue(this.localizable.getHeaderValueImport());
295            TableColumn deleteCol = this.baseTable.getColumn(
296                    this.settings.getDescDelete());
297            deleteCol.setHeaderValue(
298                    this.localizable.getHeaderValueDelete());
299    
300            if (this.isDirty()) {
301                this.setDirty(false);
302                this.fileAdmin.reloadFiles();
303                this.baseTableModel.fireTableDataChanged();
304                this.aggrTableModel.fireTableDataChanged();
305            }
306    
307            // display a label iff there are no files in the list
308            if (this.baseTableModel.getRowCount() == 0) {
309                String emptyMessage = this.localizable.getEmptyMessage(
310                        this.fileAdmin.getBaseDirectory().getAbsolutePath());
311                this.emptyLabel.setText(emptyMessage);
312                this.emptyLabel.setBackground(this.prefs.getBackground());
313                this.emptyLabel.setFont(this.prefs.getBodyFont());
314                this.viewport.setView(this.emptyLabel);
315                this.viewport.setMinimumSize(
316                        new Dimension(
317                                this.settings.getPreferredEmptyMessageWidth(),
318                                this.settings.getPreferredEmptyMessageHeight()));
319                this.viewport.setPreferredSize(
320                        new Dimension(
321                                this.settings.getPreferredEmptyMessageWidth(),
322                                this.settings.getPreferredEmptyMessageHeight()));
323                this.viewport.setMaximumSize(
324                        new Dimension(
325                                this.settings.getPreferredEmptyMessageWidth(),
326                                this.settings.getPreferredEmptyMessageHeight()));
327                return;
328            }
329    
330            this.baseTable.setBackground(this.prefs.getBackground());
331            this.baseTable.setFont(this.prefs.getBodyFont());
332            this.baseTable.setRowHeight(this.prefs.getBodyRowHeight());
333            this.scrollPane.setViewportView(this.baseTable);
334    
335            this.scrollPane.setMinimumSize(
336                    new Dimension(
337                            this.settings.getMinimumTableWidth(),
338                            this.settings.getMinimumTableHeight()));
339            this.scrollPane.setPreferredSize(
340                    new Dimension(
341                            this.prefs.getPreferredTableWidth(),
342                                    this.prefs.getPreferredTableHeight(
343                                            this.baseTableModel.getRowCount())));
344            this.scrollPane.setMaximumSize(
345                    new Dimension(
346                            this.prefs.getMaximumTableWidth(),
347                            this.prefs.getMaximumTableHeight()));
348    
349            // display only scroll pane iff there is exactly one file in the list
350            if (this.baseTableModel.getRowCount() == 1) {
351                this.scrollPane.setBorder(MoneydanceLAF.homePageBorder);
352                this.viewport.setView(this.scrollPane);
353                this.viewport.setMinimumSize(
354                        new Dimension(
355                                this.settings.getMinimumTableWidth(),
356                                this.settings.getMinimumTableHeight()));
357                this.viewport.setPreferredSize(
358                        new Dimension(
359                                this.prefs.getPreferredTableWidth(),
360                                this.prefs.getPreferredTableHeight(
361                                        this.baseTableModel.getRowCount())));
362                this.viewport.setMaximumSize(
363                        new Dimension(
364                                this.prefs.getMaximumTableWidth(),
365                                this.prefs.getMaximumTableHeight()));
366                this.registerKeyboardShortcuts4One();
367            }
368    
369            // display scroll pane and an additional import all/delete all table
370            // iff there is more than one file in the list
371            if (this.baseTableModel.getRowCount() > 1) {
372                this.scrollPane.setBorder(new EmptyBorder(0, 0, 0, 0));
373    
374                this.aggrTable.setBackground(this.prefs.getBackground());
375                this.aggrTable.setRowHeight(
376                        this.prefs.getBodyRowHeight()
377                        + this.settings.getTableHeightOffset());
378                this.aggrTable.setMinimumSize(
379                        new Dimension(
380                                this.settings.getMinimumTableWidth(),
381                                this.prefs.getBodyRowHeight()
382                                + this.settings.getTableHeightOffset()));
383                this.aggrTable.setPreferredSize(
384                        new Dimension(
385                                this.prefs.getPreferredTableWidth(),
386                                this.prefs.getBodyRowHeight()
387                                + this.settings.getTableHeightOffset()));
388                this.aggrTable.setMaximumSize(
389                        new Dimension(
390                                this.prefs.getMaximumTableWidth(),
391                                this.prefs.getBodyRowHeight()
392                                + this.settings.getTableHeightOffset()));
393    
394                this.splitPane.setTopComponent(this.scrollPane);
395                this.splitPane.setBottomComponent(this.aggrTable);
396    
397                this.viewport.setView(this.splitPane);
398                this.viewport.setMinimumSize(null);
399                this.viewport.setPreferredSize(null);
400                this.viewport.setMaximumSize(null);
401                this.registerKeyboardShortcuts4All();
402            }
403        }

406        @Override
407        public void update(final Observable observable, final Object object) {
408            this.setDirty(true);
409            this.refresh();
410        }

412    
413        @Override
414        public JComponent getGUIView(final RootAccount rootAccount) {
415            if (!this.initialized) {
416                this.initialized = true;
417                this.init();
418            }
419            return this.viewport;
420        }

425            // ignore
426        }

429        @Override
430        public void setActive(final boolean active) {
431            if (this.scrollPane == null) {
432                return;
433            }
434            this.scrollPane.setVisible(active);
435            this.refresh();
436        }

456        public void cleanup() {
457            if (!this.initialized) {
458                return;
459            }
460            this.viewport.setEnabled(false);
461            this.viewport.removeAll();
462            this.fileAdmin.stopMonitor();
463        }

468        }

473        }

476        private void registerKeyboardShortcuts4One() {
477            AbstractEditor importOneEditor =
478                this.columnFactory.getImportOneEditor();
479            importOneEditor.registerKeyboardShortcut(this.scrollPane);
480    
481            AbstractEditor deleteOneEditor =
482                this.columnFactory.getDeleteOneEditor();
483            deleteOneEditor.registerKeyboardShortcut(this.scrollPane);
484        }
485    
486    
487        private void registerKeyboardShortcuts4All() {
488            AbstractEditor importAllEditor =
489                this.columnFactory.getImportAllEditor();
490            importAllEditor.registerKeyboardShortcut(this.scrollPane);
491    
492            AbstractEditor deleteAllEditor =
493                this.columnFactory.getDeleteAllEditor();
494            deleteAllEditor.registerKeyboardShortcut(this.scrollPane);
495        }
