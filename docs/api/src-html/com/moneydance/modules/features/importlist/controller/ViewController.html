050    /**
051     * The user interface that is displayed on the homepage.
052     */
053    public final class ViewController implements HomePageView, Observer {
054    
055        /**
056         * Static initialization of class-dependent logger.
057         */
058        private static final Logger LOG = Logger.getLogger(ViewController.class);
059    
060        private final Preferences           prefs;
061        private final FileAdmin             fileAdmin;
062        private       boolean               initialized;
063        private       FileTableModel        fileTableModel;
064        private       JLabel                emptyLabel;
065        private       JTable                table;
066        private       ColumnFactory         columnFactory;
067        private       JScrollPane           scrollPane;
068        private       boolean               dirty;
069    
070    
071        public ViewController(
072                final String argBaseDirectory,
073                final FeatureModuleContext argContext) {
074            this.prefs      = Helper.getPreferences();
075            this.fileAdmin  = new FileAdmin(argBaseDirectory, argContext);
076            this.fileAdmin.addObserver(this);
077        }
078    
079    
080        private void init() {
081            this.fileTableModel = new FileTableModel(this.fileAdmin.getFiles());
082    
083            this.emptyLabel = new JLabel();
084            this.emptyLabel.setHorizontalAlignment(SwingConstants.CENTER);
085    
086            this.table = new JTable(this.fileTableModel);
087            this.table.setOpaque(false);
088            this.table.setShowGrid(false);
089            this.table.setShowVerticalLines(false);
090            this.table.setShowHorizontalLines(false);
091            this.table.setIntercellSpacing(new Dimension(0, 0));
092            this.table.setColumnSelectionAllowed(false);
093            this.table.setRowSelectionAllowed(false);
094            this.table.setCellSelectionEnabled(false);
095    
096            JTableHeader tableHeader     = this.table.getTableHeader();
097            TableColumnModel columnModel = this.table.getColumnModel();
098    
099            this.columnFactory = new ColumnFactory(
100                    this.fileAdmin,
101                    tableHeader.getDefaultRenderer());
102    
103            TableColumn nameCol = this.table.getColumn(this.prefs.getDescName());
104            int nameColNo = columnModel.getColumnIndex(this.prefs.getDescName());
105            nameCol.setIdentifier(this.prefs.getDescName());
106            nameCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
107            nameCol.setCellRenderer(
108                    this.columnFactory.getLabelNameRenderer());
109            nameCol.setMinWidth(this.prefs.getMinColumnWidth());
110            nameCol.setPreferredWidth(this.prefs.getColumnWidths(nameColNo));
111    
112            TableColumn modifiedCol = this.table.getColumn(
113                    this.prefs.getDescModified());
114            int modifiedColNo = columnModel.getColumnIndex(
115                    this.prefs.getDescModified());
116            modifiedCol.setIdentifier(this.prefs.getDescModified());
117            modifiedCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
118            modifiedCol.setCellRenderer(
119                    this.columnFactory.getLabelModifiedRenderer());
120            modifiedCol.setMinWidth(this.prefs.getMinColumnWidth());
121            modifiedCol.setPreferredWidth(
122                    this.prefs.getColumnWidths(modifiedColNo));
123    
124            TableColumn importCol = this.table.getColumn(
125                    this.prefs.getDescImport());
126            int importColNo = columnModel.getColumnIndex(
127                    this.prefs.getDescImport());
128            importCol.setIdentifier(this.prefs.getDescImport());
129            importCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
130            importCol.setCellRenderer(this.columnFactory.getButtonRenderer());
131            importCol.setCellEditor(this.columnFactory.getImportEditor());
132            importCol.setResizable(this.prefs.isButtonResizable());
133            importCol.setMinWidth(this.prefs.getMinColumnWidth());
134            importCol.setPreferredWidth(this.prefs.getColumnWidths(importColNo));
135    
136            TableColumn deleteCol = this.table.getColumn(
137                    this.prefs.getDescDelete());
138            int deleteColNo = columnModel.getColumnIndex(
139                    this.prefs.getDescDelete());
140            deleteCol.setIdentifier(this.prefs.getDescDelete());
141            deleteCol.setHeaderRenderer(this.columnFactory.getHeaderRenderer());
142            deleteCol.setCellRenderer(this.columnFactory.getButtonRenderer());
143            deleteCol.setCellEditor(this.columnFactory.getDeleteEditor());
144            deleteCol.setResizable(this.prefs.isButtonResizable());
145            deleteCol.setMinWidth(this.prefs.getMinColumnWidth());
146            deleteCol.setPreferredWidth(this.prefs.getColumnWidths(deleteColNo));
147    
148            // resizing the columns
149            TableListener tableListener = new TableListener(this.table);
150            columnModel.addColumnModelListener(tableListener);
151    
152            // reordering the columns
153            tableHeader.setReorderingAllowed(this.prefs.isReorderingAllowed());
154    
155            // sorting the columns
156            RowSorter<TableModel> rowSorter =
157                new FileTableRowSorter(this.fileTableModel);
158            List<RowSorter.SortKey> sortKeys = new ArrayList<RowSorter.SortKey>();
159            sortKeys.add(this.prefs.getSortKey());
160            rowSorter.setSortKeys(sortKeys);
161            rowSorter.addRowSorterListener(tableListener);
162            this.table.setRowSorter(rowSorter);
163    
164            this.table.setPreferredScrollableViewportSize(
165                    new Dimension(
166                            this.prefs.getMinimumTableWidth(),
167                            this.prefs.getMinimumTableHeight()));
168    
169            this.scrollPane = new JCustomScrollPane();
170            //see moneydance.com/pipermail/moneydance-dev/2006-September/000075.html
171            try {
172                this.scrollPane.setBorder(MoneydanceLAF.homePageBorder);
173            } catch (Throwable e) {
174                LOG.warn(e.getMessage(), e);
175            }
176    
177            this.setDirty(true);
178            this.refresh();
179            this.fileAdmin.startMonitor();
180        }
181    
182    
183        @Override
184        public synchronized void refresh() {
185            if (this.scrollPane == null || !this.scrollPane.isVisible()) {
186                return;
187            }
188    
189            this.prefs.reload();
190            this.columnFactory.setDateFormatter(this.prefs.getDateFormatter());
191            this.columnFactory.setTimeFormatter(this.prefs.getTimeFormatter());
192            this.columnFactory.setForeground(this.prefs.getForeground());
193            this.columnFactory.setBackground(this.prefs.getBackground());
194            this.columnFactory.setBackgroundAlt(this.prefs.getBackgroundAlt());
195            this.scrollPane.setBackground(this.prefs.getBackground());
196            TableColumn nameCol = this.table.getColumn(this.prefs.getDescName());
197            nameCol.setHeaderValue(this.prefs.getHeaderValueName());
198            TableColumn modifiedCol = this.table.getColumn(
199                    this.prefs.getDescModified());
200            modifiedCol.setHeaderValue(this.prefs.getHeaderValueModified());
201            TableColumn importCol = this.table.getColumn(
202                    this.prefs.getDescImport());
203            importCol.setHeaderValue(this.prefs.getHeaderValueImport());
204            TableColumn deleteCol = this.table.getColumn(
205                    this.prefs.getDescDelete());
206            deleteCol.setHeaderValue(
207                    this.prefs.getHeaderValueDelete());
208    
209            if (this.isDirty()) {
210                this.setDirty(false);
211                this.fileAdmin.reloadFiles();
212                this.fileTableModel.fireTableDataChanged();
213            }
214    
215            if (this.fileTableModel.getRowCount() == 0) {
216                String emptyMessage = this.prefs.getEmptyMessage(
217                        this.fileAdmin.getBaseDirectory().getAbsolutePath());
218                this.emptyLabel.setText(emptyMessage);
219                this.emptyLabel.setBackground(this.prefs.getBackground());
220                this.emptyLabel.setFont(this.prefs.getBodyFont());
221                this.scrollPane.setViewportView(this.emptyLabel);
222                this.scrollPane.setMinimumSize(
223                        new Dimension(
224                                this.prefs.getPreferredEmptyMessageWidth(),
225                                this.prefs.getPreferredEmptyMessageHeight()));
226                this.scrollPane.setPreferredSize(
227                        new Dimension(
228                                this.prefs.getPreferredEmptyMessageWidth(),
229                                this.prefs.getPreferredEmptyMessageHeight()));
230                this.scrollPane.setMaximumSize(
231                        new Dimension(
232                                this.prefs.getPreferredEmptyMessageWidth(),
233                                this.prefs.getPreferredEmptyMessageHeight()));
234            } else {
235                this.table.setBackground(this.prefs.getBackground());
236                this.table.setRowHeight(this.prefs.getBodyRowHeight());
237                this.scrollPane.setViewportView(this.table);
238                this.scrollPane.setMinimumSize(
239                        new Dimension(
240                                this.prefs.getMinimumTableWidth(),
241                                this.prefs.getMinimumTableHeight()));
242                this.scrollPane.setPreferredSize(
243                        new Dimension(
244                                this.prefs.getPreferredTableWidth(
245                                        this.fileTableModel.getRowCount()),
246                                this.prefs.getPreferredTableHeight(
247                                        this.fileTableModel.getRowCount())));
248                this.scrollPane.setMaximumSize(
249                        new Dimension(
250                                this.prefs.getMaximumTableWidth(),
251                                this.prefs.getMaximumTableHeight()));
252            }
253        }
254    
255    
256        @Override
257        public void update(final Observable observable, final Object object) {
258            this.setDirty(true);
259            this.refresh();
260        }
261    
262    
263        @Override
264        public JComponent getGUIView(final RootAccount rootAccount) {
265            if (!this.initialized) {
266                this.initialized = true;
267                this.init();
268            }
269            return this.scrollPane;
270        }
271    
272    
273        @Override
274        public void reset() {
275            // ignore
276        }
277    
278    
279        @Override
280        public void setActive(final boolean active) {
281            this.scrollPane.setVisible(active);
282            this.refresh();
283        }
284    
285    
286        @Override
287        public String getID() {
288            return this.prefs.getExtensionIdentifier();
289        }
290    
291    
292        @Override
293        public String toString() {
294            return this.prefs.getExtensionName();
295        }
296    
297    
298        public void resetBaseDirectory() {
299            this.fileAdmin.resetBaseDirectory();
300        }
301    
302    
303        public void cleanup() {
304            this.fileAdmin.stopMonitor();
305        }
306    
307    
308        public void setDirty(final boolean argDirty) {
309            this.dirty = argDirty;
310        }
311    
312    
313        public boolean isDirty() {
314            return this.dirty;
315        }
