051    /**
052     * The user interface that is displayed on the homepage.
053     *
054     * @author Florian J. Breunig
055     */
056    public final class ViewController implements HomePageView, Observer {
057    
058        private final Preferences               prefs;
059        private final Settings                  settings;
060        private final Localizable               localizable;
061        private final FileAdmin                 fileAdmin;
062        private final Tracker                   tracker;
063        private       boolean                   initialized;
064        private       ColumnFactory             columnFactory;
065        private       JViewport                 viewport;
066        private       DirectoryChooserFactory   directoryChooserFactory;
067        private       EmptyLabelFactory         emptyLabelFactory;
068        private       ComponentFactory          splitPaneFactory;
069        private       BaseTableFactory          baseTableFactory;
070        private       AbstractTableModel        baseTableModel;
071        private       AbstractTableModel        aggrTableModel;
072        private       boolean                   dirty;
073    
074    
075        public ViewController(
076                final String baseDirectory,
077                final FeatureModuleContext context,
078                final int build) {
079            this.prefs       = Helper.getPreferences();
080            this.settings    = Helper.getSettings();
081            this.localizable = Helper.getLocalizable();
082            this.fileAdmin   = new FileAdmin(baseDirectory, context);
083            this.fileAdmin.addObserver(this);
084            this.tracker     = new Tracker(
085                    this.prefs.getFullVersion(),
086                    build,
087                    this.settings.getTrackingCode());
088        }
089    
090    
091        private void init() {
092            this.viewport = new JViewport();
093            this.viewport.setOpaque(false);
094    
095            ActionListener actionListener = new ChooseBaseDirectoryActionListener();
096            this.directoryChooserFactory = new DirectoryChooserFactory(
097                    actionListener);
098            this.emptyLabelFactory       = new EmptyLabelFactory();
099    
100            this.baseTableModel   = new FileTableModel(this.fileAdmin.getFiles());
101            this.baseTableFactory = new BaseTableFactory(
102                    this.baseTableModel,
103                    this.fileAdmin);
104    
105            this.aggrTableModel   = new AggregationTableModel();
106            AggregationTableFactory aggrTableFactory = new AggregationTableFactory(
107                    this.aggrTableModel,
108                    this.fileAdmin);
109            this.columnFactory    = this.baseTableFactory.getColumnFactory();
110    
111            this.splitPaneFactory = new SplitPaneFactory(
112                    this.baseTableFactory,
113                    aggrTableFactory);
114    
115            this.setDirty(true);
116            this.refresh();
117            this.fileAdmin.startMonitor();
118    
119            this.tracker.track("Display");
120        }
121    
122    
123        @Override
124        public synchronized void refresh() {
125            if (this.viewport == null || !this.viewport.isVisible()) {
126                return;
127            }
128    
129            this.prefs.reload();
130    
131            // display a label iff no base directory has been chosen
132            if (this.fileAdmin.getBaseDirectory() == null) {
133                JButton chooserButton = this.directoryChooserFactory.getComponent();
134                chooserButton.setText(
135                        "<html><u>"
136                        + this.localizable.getDirectoryChooserTitle()
137                        + "</u></html>");
138    
139                this.viewport.setView(chooserButton);
140                this.viewport.setMinimumSize(
141                        new Dimension(
142                                this.settings.getPreferredEmptyMessageWidth(),
143                                this.settings.getPreferredEmptyMessageHeight()));
144                this.viewport.setPreferredSize(
145                        new Dimension(
146                                this.settings.getPreferredEmptyMessageWidth(),
147                                this.settings.getPreferredEmptyMessageHeight()));
148                this.viewport.setMaximumSize(
149                        new Dimension(
150                                this.settings.getPreferredEmptyMessageWidth(),
151                                this.settings.getPreferredEmptyMessageHeight()));
152                return;
153            }
154    
155            if (this.isDirty()) {
156                this.setDirty(false);
157                this.fileAdmin.reloadFiles();
158                this.baseTableModel.fireTableDataChanged();
159                this.aggrTableModel.fireTableDataChanged();
160            }
161    
162            // display a label iff there are no files in the list
163            if (this.baseTableModel.getRowCount() == 0) {
164                String emptyMessage = this.localizable.getEmptyMessage(
165                        this.fileAdmin.getBaseDirectory().getAbsolutePath());
166                JLabel emptyLabel = this.emptyLabelFactory.getComponent();
167                emptyLabel.setText(emptyMessage);
168    
169                this.viewport.setView(emptyLabel);
170                this.viewport.setMinimumSize(
171                        new Dimension(
172                                this.settings.getPreferredEmptyMessageWidth(),
173                                this.settings.getPreferredEmptyMessageHeight()));
174                this.viewport.setPreferredSize(
175                        new Dimension(
176                                this.settings.getPreferredEmptyMessageWidth(),
177                                this.settings.getPreferredEmptyMessageHeight()));
178                this.viewport.setMaximumSize(
179                        new Dimension(
180                                this.settings.getPreferredEmptyMessageWidth(),
181                                this.settings.getPreferredEmptyMessageHeight()));
182                return;
183            }
184    
185            // display only scroll pane iff there is exactly one file in the list
186            if (this.baseTableModel.getRowCount() == 1) {
187                JComponent component = this.baseTableFactory.getComponent();
188                component.setBorder(MoneydanceLAF.homePageBorder);
189    
190                this.viewport.setView(component);
191                this.viewport.setMinimumSize(
192                        new Dimension(
193                                this.settings.getMinimumTableWidth(),
194                                this.settings.getMinimumTableHeight()));
195                this.viewport.setPreferredSize(
196                        new Dimension(
197                                this.prefs.getPreferredTableWidth(),
198                                this.prefs.getPreferredTableHeight(
199                                        this.baseTableModel.getRowCount())));
200                this.viewport.setMaximumSize(
201                        new Dimension(
202                                this.prefs.getMaximumTableWidth(),
203                                this.prefs.getMaximumTableHeight()));
204                this.registerKeyboardShortcuts4One();
205                return;
206            }
207    
208            // display scroll pane and an additional import all/delete all table
209            // iff there is more than one file in the list
210            if (this.baseTableModel.getRowCount() > 1) {
211                JComponent component = this.splitPaneFactory.getComponent();
212                component.setBorder(MoneydanceLAF.homePageBorder);
213    
214                this.viewport.setView(component);
215                this.viewport.setMinimumSize(null);
216                this.viewport.setPreferredSize(null);
217                this.viewport.setMaximumSize(null);
218                this.registerKeyboardShortcuts4All();
219                return;
220            }
221        }
222    
223    
224        @Override
225        public void update(final Observable observable, final Object object) {
226            this.setDirty(true);
227            this.refresh();
228        }
229    
230    
231        @Override
232        public JComponent getGUIView(final RootAccount rootAccount) {
233            if (!this.initialized) {
234                this.initialized = true;
235                this.init();
236            }
237            return this.viewport;
238        }
239    
240    
241        @Override
242        public void reset() {
243            // ignore
244        }
245    
246    
247        @Override
248        public void setActive(final boolean active) {
249            if (this.viewport == null) {
250                return;
251            }
252            this.viewport.setVisible(active);
253            this.refresh();
254        }
255    
256    
257        @Override
258        public String getID() {
259            return this.settings.getExtensionIdentifier();
260        }
261    
262    
263        @Override
264        public String toString() {
265            return this.settings.getExtensionName();
266        }
267    
268    
269        public void chooseBaseDirectory() {
270            this.fileAdmin.chooseBaseDirectory();
271        }
272    
273    
274        public void cleanup() {
275            if (!this.initialized) {
276                return;
277            }
278            this.viewport.setEnabled(false);
279            this.viewport.removeAll();
280            this.fileAdmin.stopMonitor();
281        }
282    
283    
284        public void setDirty(final boolean argDirty) {
285            this.dirty = argDirty;
286        }
287    
288    
289        public boolean isDirty() {
290            return this.dirty;
291        }
292    
293    
294        private void registerKeyboardShortcuts4One() {
295            AbstractEditor importOneEditor =
296                this.columnFactory.getImportOneEditor();
297            importOneEditor.registerKeyboardShortcut(this.viewport);
298    
299            AbstractEditor deleteOneEditor =
300                this.columnFactory.getDeleteOneEditor();
301            deleteOneEditor.registerKeyboardShortcut(this.viewport);
302        }
303    
304    
305        private void registerKeyboardShortcuts4All() {
306            AbstractEditor importAllEditor =
307                this.columnFactory.getImportAllEditor();
308            importAllEditor.registerKeyboardShortcut(this.viewport);
309    
310            AbstractEditor deleteAllEditor =
311                this.columnFactory.getDeleteAllEditor();
312            deleteAllEditor.registerKeyboardShortcut(this.viewport);
313        }
314    
315    
316        private class ChooseBaseDirectoryActionListener implements ActionListener {
317    
318            @Override
319            public void actionPerformed(final ActionEvent event) {
320                ViewController.this.chooseBaseDirectory();
321            }
322        }
