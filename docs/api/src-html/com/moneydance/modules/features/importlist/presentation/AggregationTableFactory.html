017package com.moneydance.modules.features.importlist.presentation;
018
019import com.moneydance.modules.features.importlist.io.FileAdmin;
020import com.moneydance.modules.features.importlist.table.ColumnFactory;
021import com.moneydance.modules.features.importlist.util.Helper;
022import com.moneydance.modules.features.importlist.util.Preferences;
023import com.moneydance.modules.features.importlist.util.Settings;
024
025import java.awt.Dimension;
026
027import javax.swing.JTable;
028import javax.swing.table.JTableHeader;
029import javax.swing.table.TableColumn;
030import javax.swing.table.TableColumnModel;
031import javax.swing.table.TableModel;
032
033/**
034 * @author Florian J. Breunig
035 */
036public final class AggregationTableFactory implements ComponentFactory {
037
038    private final Settings settings;
039    private final Preferences prefs;
040    private final JTable table;
041
042    public AggregationTableFactory(
043            final TableModel argTableModel,
044            final FileAdmin argFileAdmin) {
045        this.settings   = Helper.INSTANCE.getSettings();
046        this.prefs      = Helper.INSTANCE.getPreferences();
047
048        this.table = new JTable(argTableModel);
049        this.table.setOpaque(false);
050        this.table.setShowGrid(false);
051        this.table.setShowVerticalLines(false);
052        this.table.setShowHorizontalLines(false);
053        this.table.setIntercellSpacing(
054                new Dimension(
055                        0,
056                        this.settings.getTableHeightOffset()));
057        this.table.setColumnSelectionAllowed(false);
058        this.table.setRowSelectionAllowed(false);
059        this.table.setCellSelectionEnabled(false);
060
061        TableColumnModel aggrColumnModel = this.table.getColumnModel();
062        JTableHeader tableHeader         = this.table.getTableHeader();
063
064        ColumnFactory columnFactory = new ColumnFactory(
065                argFileAdmin,
066                tableHeader.getDefaultRenderer(),
067                this.prefs.getDateFormatter(),
068                Preferences.getTimeFormatter());
069
070
071        TableColumn nameCol = this.table.getColumn(this.settings.getDescName());
072        int nameColNo = aggrColumnModel.getColumnIndex(
073                this.settings.getDescName());
074        nameCol.setIdentifier(this.settings.getDescName());
075        nameCol.setCellRenderer(
076                columnFactory.getLabelNameAllRenderer());
077        nameCol.setMinWidth(this.settings.getMinColumnWidth());
078        nameCol.setPreferredWidth(this.prefs.getColumnWidths(nameColNo));
079
080        TableColumn modifiedCol = this.table.getColumn(
081                this.settings.getDescModified());
082        int modifiedColNo = aggrColumnModel.getColumnIndex(
083                this.settings.getDescModified());
084        modifiedCol.setIdentifier(this.settings.getDescModified());
085        modifiedCol.setCellRenderer(
086                columnFactory.getLabelModifiedAllRenderer());
087        modifiedCol.setMinWidth(this.settings.getMinColumnWidth());
088        modifiedCol.setPreferredWidth(
089                this.prefs.getColumnWidths(modifiedColNo));
090
091        TableColumn importCol = this.table.getColumn(
092                this.settings.getDescImport());
093        int importColNo = aggrColumnModel.getColumnIndex(
094                this.settings.getDescImport());
095        importCol.setIdentifier(this.settings.getDescImport());
096        importCol.setCellRenderer(columnFactory.getButtonAllRenderer());
097        importCol.setCellEditor(columnFactory.getImportAllEditor());
098        importCol.setResizable(this.settings.isButtonResizable());
099        importCol.setMinWidth(this.settings.getMinColumnWidth());
100        importCol.setPreferredWidth(this.prefs.getColumnWidths(importColNo));
101
102        TableColumn deleteCol = this.table.getColumn(
103                this.settings.getDescDelete());
104        int deleteColNo = aggrColumnModel.getColumnIndex(
105                this.settings.getDescDelete());
106        deleteCol.setIdentifier(this.settings.getDescDelete());
107        deleteCol.setCellRenderer(columnFactory.getButtonAllRenderer());
108        deleteCol.setCellEditor(columnFactory.getDeleteAllEditor());
109        deleteCol.setResizable(this.settings.isButtonResizable());
110        deleteCol.setMinWidth(this.settings.getMinColumnWidth());
111        deleteCol.setPreferredWidth(this.prefs.getColumnWidths(deleteColNo));
112    }
113
114    @Override
115    public JTable getComponent() {
116        this.table.setRowHeight(
117                this.prefs.getBodyRowHeight()
118                + this.settings.getTableHeightOffset());
119        this.table.setMinimumSize(
120                new Dimension(
121                        this.settings.getMinimumTableWidth(),
122                        this.prefs.getBodyRowHeight()
123                        + this.settings.getTableHeightOffset()));
124        this.table.setPreferredSize(
125                new Dimension(
126                        this.prefs.getPreferredTableWidth(),
127                        this.prefs.getBodyRowHeight()
128                        + this.settings.getTableHeightOffset()));
129        this.table.setMaximumSize(
130                new Dimension(
131                        Preferences.getMaximumTableWidth(),
132                        this.prefs.getBodyRowHeight()
133                        + this.settings.getTableHeightOffset()));
134
135        return this.table;
136    }
137}
