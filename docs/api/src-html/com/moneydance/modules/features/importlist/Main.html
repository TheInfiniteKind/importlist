001    package com.moneydance.modules.features.importlist;
002    
003    import java.awt.Image;
004    
005    import org.apache.log4j.Logger;
006    
007    import com.moneydance.apps.md.controller.FeatureModule;
008    import com.moneydance.apps.md.view.HomePageView;
009    import com.moneydance.modules.features.importlist.io.FileAdministration;
010    import com.moneydance.modules.features.importlist.util.Preferences;
011    import com.moneydance.modules.features.importlist.view.View;
012    
013    /**
014     * The main class of the extension, instantiated by Moneydance's class
015     * loader.
016     *
017     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
018     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
019     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
020     */
021    public class Main extends FeatureModule {
022    
023        /**
024         * Static initialization of class-dependent logger.
025         */
026        private static Logger log = Logger.getLogger(Main.class);
027    
028        private String             baseDirectory;
029        private FileAdministration fileAdministration;
030        private HomePageView       homePageView;
031        private final Preferences  prefs;
032    
033        static {
034            Preferences.loadLoggerConfiguration();
035        }
036    
037        /**
038         * Standard constructor must be available in the Moneydance context.
039         */
040        public Main() {
041            super();
042            log.info("Initializing extension in Moneydance's application context.");
043            this.prefs = new Preferences(this);
044        }
045    
046        public Main(final String argBaseDirectory) {
047            super();
048            log.info("Initializing extension in stand-alone mode.");
049            this.baseDirectory = argBaseDirectory;
050            this.prefs = new Preferences(this);
051        }
052    
053        @Override
054        public final void init() {
055            this.fileAdministration = new FileAdministration(
056                    this,
057                    this.baseDirectory);
058            this.homePageView       = new View(this.fileAdministration);
059            this.fileAdministration.setHomePageView(this.homePageView);
060    
061            if (this.getContext() != null) {
062                log.debug("Registering homepage view");
063                // register this module's homepage view
064                this.getContext().registerHomePageView(this, this.homePageView);
065    
066                log.debug("Registering toolbar feature");
067                // register this module to be invoked via the application toolbar
068                this.getContext().registerFeature(
069                        this,
070                        this.prefs.getChooseBaseDirSuffix(),
071                        null,
072                        this.getName());
073            }
074        }
075    
076        @Override
077        public final String getName() {
078            return this.prefs.getExtensionName();
079        }
080    
081        @Override
082        public final Image getIconImage() {
083            return this.prefs.getIconImage();
084        }
085    
086        @Override
087        public final void invoke(final String uri) {
088            if (this.prefs.getChooseBaseDirSuffix().equals(uri)) {
089                this.fileAdministration.reset();
090            }
091    
092            if (this.prefs.getReloadContextSuffix().equals(uri)) {
093                log.info("Reloading context from underlying framework.");
094                if (this.fileAdministration != null) {
095                    this.fileAdministration.setContext(this.getContext());
096                }
097                this.prefs.setContext(this.getContext());
098            }
099        }
100    
101        @Override
102        public final void unload() {
103            log.info("Unloading extension.");
104            this.cleanup();
105            this.homePageView.setActive(false);
106            this.prefs.setAllWritablePreferencesToNull();
107        }
108    
109        @Override
110        public final void cleanup() {
111            this.fileAdministration.stopMonitor();
112        }
113    
114        public final HomePageView getHomePageView() {
115            return this.homePageView;
116        }
117    }
