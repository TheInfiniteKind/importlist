017package com.moneydance.modules.features.importlist;
018
019import com.moneydance.apps.md.controller.FeatureModule;
020import com.moneydance.apps.md.view.HomePageView;
021import com.moneydance.modules.features.importlist.controller.ViewController;
022import com.moneydance.modules.features.importlist.util.Helper;
023import com.moneydance.modules.features.importlist.util.Preferences;
024import com.moneydance.modules.features.importlist.util.Settings;
025import com.moneydance.modules.features.importlist.util.Tracker;
026
027import java.awt.Image;
028import java.util.Observable;
029import java.util.Observer;
030import java.util.logging.Logger;
031
032/**
033 * The main class of the extension, instantiated by Moneydance's class loader.
034 *
035 * @author Florian J. Breunig
036 */
037public final class Main extends FeatureModule implements Observer {
038
039    /**
040     * Static initialization of class-dependent logger.
041     */
042    private static final Logger LOG = Logger.getLogger(Main.class.getName());
043
044    private final Preferences     prefs;
045    private final Settings       settings;
046    private       Tracker         tracker;
047    private       String          baseDirectory;
048    private       ViewController  viewController;
049
050    static {
051        Helper.loadLoggerConfiguration();
052    }
053
054    /**
055     * Public standard constructor must be available in the Moneydance context.
056     */
057    public Main() {
058        LOG.info("Initializing extension in Moneydance's application context.");
059        this.prefs = Helper.INSTANCE.getPreferences();
060        this.settings = Helper.INSTANCE.getSettings();
061    }
062
063    @Override
064    public void init() {
065        Helper.INSTANCE.addObserver(this);
066        this.tracker  = Helper.INSTANCE.getTracker(this.getBuild());
067
068        if (this.prefs.isFirstRun()) {
069            this.prefs.setFirstRun(false);
070            LOG.config("Install");
071            this.tracker.track(Tracker.EventName.INSTALL);
072        }
073
074        this.viewController = new ViewController(
075                this.baseDirectory,
076                this.getContext(),
077                this.tracker);
078
079        // register this module's homepage view
080        LOG.config("Registering homepage view");
081        this.getContext().registerHomePageView(this, this.viewController);
082
083        // register this module to be invoked via the application toolbar
084        LOG.config("Registering toolbar feature");
085        this.getContext().registerFeature(
086                this,
087                this.settings.getChooseBaseDirSuffix(),
088                null, // buttonImage
089                this.getName());
090    }
091
092    @Override
093    public String getName() {
094        return this.settings.getExtensionName();
095    }
096
097    @Override
098    public Image getIconImage() {
099        return this.settings.getIconImage();
100    }
101
102    @Override
103    public void invoke(final String uri) {
104        if (this.settings.getChooseBaseDirSuffix().equals(uri)) {
105            this.viewController.chooseBaseDirectory();
106        }
107    }
108
109    @Override
110    public void update(final Observable observable, final Object updateAll) {
111        LOG.info("Reloading context from underlying framework.");
112        Helper.INSTANCE.setContext(this.getContext());
113    }
114
115    @Override
116    public void unload() {
117        LOG.info("Unloading extension."); //$NON-NLS-1$
118        this.tracker.track(Tracker.EventName.UNINSTALL);
119        this.viewController.setActive(false);
120        this.cleanup();
121        this.prefs.setAllWritablePreferencesToNull();
122    }
123
124    @Override
125    public void cleanup() {
126        this.viewController.cleanup();
127    }
128
129    void setBaseDirectory(final String argBaseDirectory) {
130        this.baseDirectory = argBaseDirectory;
131    }
132
133    HomePageView getHomePageView() {
134        return this.viewController;
135    }
136}
