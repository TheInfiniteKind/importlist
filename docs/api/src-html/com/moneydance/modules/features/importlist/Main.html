019    package com.moneydance.modules.features.importlist;
020    
021    import java.awt.Image;
022    import java.io.File;
023    
024    import org.apache.log4j.Logger;
025    
026    import com.moneydance.apps.md.controller.FeatureModule;
027    import com.moneydance.apps.md.controller.StubContextFactory;
028    import com.moneydance.apps.md.view.HomePageView;
029    import com.moneydance.modules.features.importlist.io.FileAdmin;
030    import com.moneydance.modules.features.importlist.util.Preferences;
031    import com.moneydance.modules.features.importlist.view.View;
032    
033    /**
034     * The main class of the extension, instantiated by Moneydance's class
035     * loader.
036     */
037    public class Main extends FeatureModule {
038    
039        static {
040            Preferences.loadLoggerConfiguration();
041        }
042    
043        /**
044         * Static initialization of class-dependent logger.
045         */
046        private static Logger log = Logger.getLogger(Main.class);
047    
048        private String             baseDirectory;
049        private FileAdmin          fileAdmin;
050        private HomePageView       homePageView;
051        private final Preferences  prefs;
052    
053        /**
054         * Standard constructor must be available in the Moneydance context.
055         */
056        public Main() {
057            super();
058            log.info("Initializing extension in Moneydance's application context.");
059            this.prefs = new Preferences(this);
060        }
061    
062        public Main(final String argBaseDirectory) {
063            super();
064            log.info("Initializing extension in stand-alone mode.");
065            this.baseDirectory = argBaseDirectory;
066            this.prefs = new Preferences(this);
067        }
068    
069        @Override
070        public final void init() {
071            StubContextFactory stubContextFactory =
072                new StubContextFactory(this, this.getContext());
073            stubContextFactory.setup();
074    
075            this.fileAdmin = new FileAdmin(this, this.baseDirectory);
076            this.homePageView = new View(this.fileAdmin);
077            this.fileAdmin.setHomePageView(this.homePageView);
078    
079            log.debug("Registering homepage view");
080            // register this module's homepage view
081            this.getContext().registerHomePageView(this, this.homePageView);
082    
083            log.debug("Registering toolbar feature");
084            // register this module to be invoked via the application toolbar
085            this.getContext().registerFeature(
086                    this,
087                    this.prefs.getChooseBaseDirSuffix(),
088                    null,
089                    this.getName());
090        }
091    
092        @Override
093        public final String getName() {
094            return this.prefs.getExtensionName();
095        }
096    
097        @Override
098        public final Image getIconImage() {
099            return this.prefs.getIconImage();
100        }
101    
102        @Override
103        public final void invoke(final String uri) {
104            String command = uri;
105            String parameters = "";
106            int theIdx = uri.indexOf('?');
107            if (theIdx >= 0) {
108                command = uri.substring(0, theIdx);
109                parameters = uri.substring(theIdx + 1);
110            } else {
111                theIdx = uri.indexOf(':');
112                if (theIdx >= 0) {
113                    command = uri.substring(0, theIdx);
114                }
115            }
116    
117            if (this.prefs.getChooseBaseDirSuffix().equals(command)) {
118                this.fileAdmin.reset();
119            }
120    
121            if (this.prefs.getReloadContextSuffix().equals(command)) {
122                log.info("Reloading context from underlying framework.");
123                if (this.fileAdmin != null) {
124                    this.fileAdmin.setContext(this.getContext());
125                }
126                this.prefs.setContext(this.getContext());
127            }
128    
129            if (this.prefs.getDeleteFileSuffix().equals(command)) {
130                File deleteFile = new File(parameters);
131                this.fileAdmin.deleteFile(deleteFile);
132            }
133        }
134    
135        @Override
136        public final void unload() {
137            log.info("Unloading extension.");
138            this.homePageView.setActive(false);
139            this.cleanup();
140            this.prefs.setAllWritablePreferencesToNull();
141        }
142    
143        @Override
144        public final void cleanup() {
145            this.fileAdmin.stopMonitor();
146        }
147    
148        public final HomePageView getHomePageView() {
149            return this.homePageView;
150        }
151    }
