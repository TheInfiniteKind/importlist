001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.awt.Component;
004    
005    import javax.swing.AbstractButton;
006    import javax.swing.BorderFactory;
007    import javax.swing.JButton;
008    import javax.swing.JTable;
009    import javax.swing.border.Border;
010    import javax.swing.border.EtchedBorder;
011    import javax.swing.table.TableCellRenderer;
012    
013    /**
014     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
015     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
016     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
017     */
018    class ButtonRenderer implements TableCellRenderer {
019    
020        private static final long serialVersionUID = -5002244293774549298L;
021    
022        /**
023         * Standard constructor to restrict visibility.
024         */
025        ButtonRenderer() {
026        }
027    
028        @Override
029        public final Component getTableCellRendererComponent(
030                final JTable table,
031                final Object value,
032                final boolean isSelected,
033                final boolean hasFocus,
034                final int row,
035                final int column) {
036            return this.getTableCellRendererButton(
037                    table, value, hasFocus, hasFocus, row, column);
038        }
039    
040        final AbstractButton getTableCellRendererButton(
041                final JTable table,
042                final Object value,
043                final boolean isSelected,
044                final boolean hasFocus,
045                final int row,
046                final int column) {
047            JButton button = new JButton();
048            button.setOpaque(false);
049            ColorSchemeHelper.applyColorScheme(button, row);
050            Border border = BorderFactory.createEtchedBorder(
051                    EtchedBorder.LOWERED,
052                    button.getBackground().brighter(),
053                    button.getBackground().darker());
054            if (row % 2 == 0) {
055                border = BorderFactory.createEtchedBorder(
056                        EtchedBorder.RAISED,
057                        button.getBackground().darker(),
058                        button.getBackground().brighter());
059            }
060            button.setBorder(border);
061            if (value != null) {
062                button.setText(value.toString());
063            }
064            return button;
065        }
066    }
