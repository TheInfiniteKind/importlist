001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.awt.Component;
004    
005    import javax.swing.AbstractButton;
006    import javax.swing.JButton;
007    import javax.swing.JTable;
008    import javax.swing.table.TableCellRenderer;
009    
010    import com.moneydance.apps.md.view.gui.MoneydanceLAF;
011    
012    /**
013     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
014     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
015     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
016     */
017    public class ButtonRenderer implements TableCellRenderer {
018    
019        private static final long serialVersionUID = -5002244293774549298L;
020    
021        /**
022         * Standard constructor to restrict the visibility.
023         */
024        ButtonRenderer() {
025        }
026    
027        public final Component getTableCellRendererComponent(
028                final JTable table,
029                final Object value,
030                final boolean isSelected,
031                final boolean hasFocus,
032                final int row,
033                final int column) {
034            return this.getTableCellRendererButton(
035                    table, value, hasFocus, hasFocus, row, column);
036        }
037    
038        final AbstractButton getTableCellRendererButton(
039                final JTable table,
040                final Object value,
041                final boolean isSelected,
042                final boolean hasFocus,
043                final int row,
044                final int column) {
045            JButton button = new JButton();
046            button.setOpaque(true);
047            ColorSchemeHelper.applyColorScheme(button, row);
048            //see moneydance.com/pipermail/moneydance-dev/2006-September/000075.html
049            try {
050                button.setBorder(MoneydanceLAF.homePageBorder);
051            } catch (Throwable e) {
052                e.printStackTrace(System.err);
053            }
054            if (value != null) {
055                button.setText(value.toString());
056            }
057            return button;
058        }
059    }
