001    package com.moneydance.modules.features.importlist.io;
002    
003    import java.io.File;
004    
005    import org.apache.commons.io.monitor.FileAlterationListenerAdaptor;
006    import org.apache.commons.io.monitor.FileAlterationObserver;
007    import org.apache.commons.lang.Validate;
008    
009    /**
010     * This <code>FileAlterationListener</code> implementation is notified about
011     * relevant modifications in the file system. It propagates these modifications
012     * to the <code>FileAdministration</code>.
013     *
014     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
015     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
016     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
017     */
018    public class TransactionFileListener extends FileAlterationListenerAdaptor {
019    
020        private final FileAdministration fileAdministration;
021        private boolean dirty;
022    
023        TransactionFileListener(final FileAdministration argFileAdministration) {
024            Validate.notNull(argFileAdministration,
025            "argFileAdministration can't be null");
026            this.fileAdministration = argFileAdministration;
027        }
028    
029        @Override
030        public final void onStart(final FileAlterationObserver observer) {
031            this.dirty = false;
032        }
033    
034        @Override
035        public final void onFileCreate(final File file) {
036            this.dirty = true;
037        }
038    
039        @Override
040        public final void onFileChange(final File file) {
041            this.dirty = true;
042        }
043    
044        @Override
045        public final void onFileDelete(final File file) {
046            this.dirty = true;
047        }
048    
049        @Override
050        public final void onStop(final FileAlterationObserver observer) {
051            if (this.dirty) {
052                this.fileAdministration.setDirty(true);
053            }
054        }
055    }
