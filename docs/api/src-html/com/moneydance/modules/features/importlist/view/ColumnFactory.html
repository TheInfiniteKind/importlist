001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.awt.Color;
004    
005    import javax.swing.table.TableCellEditor;
006    import javax.swing.table.TableCellRenderer;
007    
008    import org.apache.commons.lang.Validate;
009    
010    import com.moneydance.modules.features.importlist.io.FileAdministration;
011    
012    /**
013     * This factory provides unified access to the required renderer classes.
014     *
015     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
016     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
017     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
018     */
019    public class ColumnFactory {
020    
021        private final HeaderRenderer    headerRenderer;
022        private final LabelRenderer     labelRenderer;
023        private final ButtonRenderer    buttonRenderer;
024        private final ImportEditor      importEditor;
025        private final DeleteEditor      deleteEditor;
026    
027        ColumnFactory(final FileAdministration argFileAdministration,
028                final TableCellRenderer argDefaultHeaderTableCellRenderer) {
029            Validate.notNull(argFileAdministration,
030            "argFileAdministration can't be null");
031            this.headerRenderer = new HeaderRenderer(
032                    argDefaultHeaderTableCellRenderer);
033            this.labelRenderer  = new LabelRenderer();
034            this.buttonRenderer = new ButtonRenderer();
035            this.importEditor   = new ImportEditor(
036                    argFileAdministration,
037                    this.buttonRenderer);
038            this.deleteEditor   = new DeleteEditor(
039                    argFileAdministration,
040                    this.buttonRenderer);
041        }
042    
043        final TableCellRenderer getHeaderRenderer() {
044            return this.headerRenderer;
045        }
046    
047        final TableCellRenderer getLabelRenderer() {
048            return this.labelRenderer;
049        }
050    
051        final TableCellRenderer getButtonRenderer() {
052            return this.buttonRenderer;
053        }
054    
055        final TableCellEditor getImportEditor() {
056            return this.importEditor;
057        }
058    
059        final TableCellEditor getDeleteEditor() {
060            return this.deleteEditor;
061        }
062    
063        final void setForegroundColor(final Color argForegroundColor) {
064            ColorSchemeHelper.setForegroundColor(argForegroundColor);
065        }
066    
067        final void setBackgroundColor(final Color argBackgroundColor) {
068            ColorSchemeHelper.setBackgroundColor(argBackgroundColor);
069        }
070    
071        final void setBackgroundColorAlt(final Color argBackgroundColorAlt) {
072            ColorSchemeHelper.setBackgroundColorAlt(argBackgroundColorAlt);
073        }
074    }
