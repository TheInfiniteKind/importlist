019    package com.moneydance.modules.features.importlist.view;
020    
021    import java.awt.Color;
022    import java.text.DateFormat;
023    
024    import javax.swing.table.TableCellEditor;
025    import javax.swing.table.TableCellRenderer;
026    
027    import org.apache.commons.lang.Validate;
028    
029    import com.moneydance.modules.features.importlist.io.FileAdmin;
030    import com.moneydance.modules.features.importlist.util.Helper;
031    import com.moneydance.modules.features.importlist.util.Preferences;
032    import com.moneydance.util.CustomDateFormat;
033    
034    /**
035     * This factory provides unified access to the required renderer classes.
036     */
037    public final class ColumnFactory {
038    
039        private final ColorSchemeHelper     colorSchemeHelper;
040        private final HeaderRenderer        headerRenderer;
041        private final LabelNameRenderer     labelNameRenderer;
042        private final LabelModifiedRenderer labelModifiedRenderer;
043        private final ButtonRenderer        buttonRenderer;
044        private final ImportEditor          importEditor;
045        private final DeleteEditor          deleteEditor;
046    
047        public ColumnFactory(
048                final FileAdmin argFileAdmin,
049                final TableCellRenderer argDefaultHeaderTableCellRenderer) {
050            Validate.notNull(argFileAdmin, "argFileAdmin can't be null");
051            Preferences prefs          = Helper.getPreferences();
052    
053            int opaqueVersion = prefs.getVersionWithOpaqueHomepageView();
054            if (prefs.getVersion() >= opaqueVersion) {
055                this.colorSchemeHelper = new OpaqueColorSchemeHelper();
056            } else {
057                this.colorSchemeHelper = new DefaultColorSchemeHelper();
058            }
059            this.headerRenderer         = new HeaderRenderer(
060                    this.colorSchemeHelper,
061                    argDefaultHeaderTableCellRenderer);
062            this.labelNameRenderer      = new LabelNameRenderer(
063                    this.colorSchemeHelper);
064            this.labelModifiedRenderer  = new LabelModifiedRenderer(
065                    this.colorSchemeHelper);
066            this.buttonRenderer         = new ButtonRenderer(
067                    this.colorSchemeHelper);
068            this.importEditor           = new ImportEditor(
069                    argFileAdmin,
070                    this.buttonRenderer);
071            this.deleteEditor           = new DeleteEditor(
072                    argFileAdmin,
073                    this.buttonRenderer);
074        }
075    
076        public TableCellRenderer getHeaderRenderer() {
077            return this.headerRenderer;
078        }
079    
080        public TableCellRenderer getLabelNameRenderer() {
081            return this.labelNameRenderer;
082        }
083    
084        public TableCellRenderer getLabelModifiedRenderer() {
085            return this.labelModifiedRenderer;
086        }
087    
088        public TableCellRenderer getButtonRenderer() {
089            return this.buttonRenderer;
090        }
091    
092        public TableCellEditor getImportEditor() {
093            return this.importEditor;
094        }
095    
096        public TableCellEditor getDeleteEditor() {
097            return this.deleteEditor;
098        }
099    
100        public void setForeground(final Color foreground) {
101            this.colorSchemeHelper.setForeground(foreground);
102        }
103    
104        public void setBackground(final Color background) {
105            this.colorSchemeHelper.setBackground(background);
106        }
107    
108        public void setBackgroundAlt(final Color backgroundAlt) {
109            this.colorSchemeHelper.setBackgroundAlt(backgroundAlt);
110        }
111    
112        public void setDateFormatter(final CustomDateFormat argDateFormatter) {
113            this.labelModifiedRenderer.setDateFormatter(argDateFormatter);
114        }
115    
116        public void setTimeFormatter(final DateFormat argTimeFormatter) {
117            this.labelModifiedRenderer.setTimeFormatter(argTimeFormatter);
118        }
119    }
