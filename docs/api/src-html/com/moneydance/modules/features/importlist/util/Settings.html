019    package com.moneydance.modules.features.importlist.util;
020    
021    import java.io.InputStream;
022    
023    import org.apache.commons.configuration.AbstractFileConfiguration;
024    import org.apache.commons.configuration.Configuration;
025    import org.apache.commons.configuration.ConfigurationException;
026    import org.apache.commons.configuration.PropertiesConfiguration;
027    import org.slf4j.Logger;
028    import org.slf4j.LoggerFactory;
029    
030    /**
031     * This configuration class accesses all values that are read
032     * from a settings file in plain text. The settings file cannot be
033     * modified at runtime, so all <code>Settings</code> objects are immutable.
034     *
035     * @author Florian J. Breunig
036     */
037    public final class Settings {
038    
039        /**
040         * Static initialization of class-dependent logger.
041         */
042        private static final Logger LOG =
043                LoggerFactory.getLogger(Settings.class);
044    
045        private final Configuration config;
046    
047        /**
048         * The resource in the JAR file to read the settings from.
049         */
050        private static final String PROPERTIES_RESOURCE
051        = "com/moneydance/modules/features/importlist/resources/"
052                + "settings.properties";
053    
054        Settings() {
055            final AbstractFileConfiguration abstractFileConfiguration =
056                    new PropertiesConfiguration();
057    
058            try {
059                InputStream inputStream = Helper.getInputStreamFromResource(
060                        PROPERTIES_RESOURCE);
061                abstractFileConfiguration.load(inputStream);
062            } catch (IllegalArgumentException e) {
063                LOG.warn(e.getMessage(), e);
064            } catch (ConfigurationException e) {
065                LOG.warn(e.getMessage(), e);
066            }
067            this.config = abstractFileConfiguration;
068        }
069    
070        /**
071         * @return The descriptive name of this extension.
072         */
073        public String getExtensionName() {
074            return this.config.getString("extension_name");
075        }
076    
077        /**
078         * @return The ID string for this extension.
079         */
080        public String getExtensionIdentifier() {
081            return this.config.getString("extension_identifier");
082        }
083    
084        /**
085         * @return The icon image that represents this extension.
086         */
087        public String getIconResource() {
088            return this.config.getString("icon_resource");
089        }
090    
091        /**
092         * @return The resource that contains the configuration of log4j.
093         */
094        public String getLog4jPropertiesResource() {
095            return this.config.getString("log4j_properties_resource");
096        }
097    
098        /**
099         * @return The resource in the JAR file to read the language strings from.
100         */
101        public String getLocalizableResource() {
102            return this.config.getString("localizable_resource");
103        }
104    
105        /**
106         * @return The suffix of the application event that lets the user change the
107         * base directory.
108         */
109        public String getChooseBaseDirSuffix() {
110            return this.config.getString("choose_base_dir_suffix");
111        }
112    
113        /**
114         * @return The prefix of the application event that imports a given file.
115         */
116        public String getTransactionFileImportUriPrefix() {
117            return this.config.getString("transaction_file_import_uri_prefix");
118        }
119

123         */

134        }

329        }
330    
331        /**
332         * @return Keyboard shortcut to delete files.
333         */
334        public String getKeyboardShortcutDelete() {
335            return this.config.getString("keyboard_shortcut_delete");

337    
338        /**
339         * @return Tracking code for Google Analytics (aka "utmac").
340         */
341        public String getTrackingCode() {
342            return this.config.getString("tracking_code");
343        }
