001    package com.moneydance.modules.features.importlist.view;
002    
003    import java.io.File;
004    import java.text.DateFormat;
005    import java.util.Date;
006    import java.util.Hashtable;
007    import java.util.List;
008    import java.util.Map;
009    
010    import javax.swing.table.AbstractTableModel;
011    
012    import org.apache.commons.lang.Validate;
013    
014    import com.moneydance.modules.features.importlist.Constants;
015    import com.moneydance.modules.features.importlist.Preferences;
016    import com.moneydance.util.CustomDateFormat;
017    
018    /**
019     * This class provides a <code>TableModel</code> implementation for a given
020     * <code>List</code> of <code>File</code>s. It takes care of the formatting and
021     * caching of its table values. The first two columns represent the name of the
022     * file and the date of its last modification, the last two columns represent
023     * the action buttons to import and delete the file, respectively.
024     *
025     * @author <a href="mailto:&#102;&#108;&#111;&#114;&#105;&#97;&#110;&#46;&#106;
026     *&#46;&#98;&#114;&#101;&#117;&#110;&#105;&#103;&#64;&#109;&#121;&#45;&#102;
027     *&#108;&#111;&#119;&#46;&#99;&#111;&#109;">Florian J. Breunig</a>
028     */
029    public class ListTableModel extends AbstractTableModel {
030    
031        private static final long serialVersionUID = 3552703741263935211L;
032        private final List<File>             files;
033        private final Map<Integer, String>   fileDateStringCache;
034        private CustomDateFormat             dateFormatter;
035        private DateFormat                   timeFormatter;
036    
037        ListTableModel(final List<File> argFiles) {
038            Validate.notNull(argFiles, "argFiles can't be null");
039            this.files               = argFiles;
040            this.fileDateStringCache = new Hashtable<Integer, String>();
041        }
042    
043        @Override
044        public final String getValueAt(final int row, final int column) {
045            String columnName = this.getColumnName(column);
046    
047            if (Constants.DESC_NAME.equals(columnName)) {
048                File file = this.getFileAt(row);
049                return file.getName();
050            }
051    
052            if (Constants.DESC_MODIFIED.equals(columnName)) {
053                if (this.fileDateStringCache.get(row) == null) {
054                    File file             = this.getFileAt(row);
055                    Date fileDate         = new Date(file.lastModified());
056                    String fileDateString = this.dateFormatter.format(fileDate)
057                    + " " + this.timeFormatter.format(fileDate);
058    
059                    this.fileDateStringCache.put(row, fileDateString);
060                }
061                return this.fileDateStringCache.get(row);
062            }
063    
064            if (Constants.DESC_IMPORT.equals(columnName)) {
065                return Constants.LABEL_IMPORT_BUTTON;
066            }
067    
068            if (Constants.DESC_DELETE.equals(columnName)) {
069                return Constants.LABEL_DELETE_BUTTON;
070            }
071    
072            return null;
073        }
074    
075    
076        @Override
077        public final void fireTableDataChanged() {
078            super.fireTableDataChanged();
079            this.fileDateStringCache.clear();
080        }
081    
082        /**
083         * Only the last two columns contain buttons and are therefore editable.
084         *
085         * @param row    the row whose value is to be queried
086         * @param column the column whose value is to be queried
087         * @return true  if the column is the last or penultimate one in the model
088         */
089        @Override
090        public final boolean isCellEditable(final int row, final int column) {
091            String columnName = this.getColumnName(column);
092            return Constants.DESC_IMPORT.equals(columnName)
093            || Constants.DESC_DELETE.equals(columnName);
094        }
095    
096        @Override
097        public final int getColumnCount() {
098            return Preferences.getInstance().getColumnCount();
099        }
100    
101        @Override
102        public final String getColumnName(final int column) {
103            return Preferences.getInstance().getColumnName(column);
104        }
105    
106        @Override
107        public final int getRowCount() {
108            if (this.files == null) {
109                return 0;
110            }
111            return this.files.size();
112        }
113    
114        final File getFileAt(final int row) {
115            return this.files.get(row);
116        }
117    
118        final void setDateFormatter(
119                final CustomDateFormat argDateFormatter) {
120            Validate.notNull(argDateFormatter, "argDateFormatter can't be null");
121            if (this.dateFormatter != null
122                    && !argDateFormatter.equals(this.dateFormatter)) {
123                this.fileDateStringCache.clear();
124            }
125            this.dateFormatter = argDateFormatter;
126        }
127    
128        final void setTimeFormatter(final DateFormat argTimeFormatter) {
129            Validate.notNull(argTimeFormatter, "argTimeFormatter can't be null");
130            if (this.timeFormatter != null
131                    && !argTimeFormatter.equals(this.timeFormatter)) {
132                this.fileDateStringCache.clear();
133            }
134            this.timeFormatter = argTimeFormatter;
135        }
136    }
