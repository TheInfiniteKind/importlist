061        /**
062         * The constructor must be called exactly once before using the only
063         * instance of this class.
064         * @param argFeatureModule The feature module to receive the context from.
065         */
066        public Preferences(final FeatureModule argFeatureModule) {
067            Validate.isTrue(instance == null,
068            "You can instantiate Preferences only once");
069            Validate.notNull(argFeatureModule, "argFeatureModule can't be null");
070    
071            AbstractFileConfiguration abstractFileConfiguration =
072                new PropertiesConfiguration();
073    
074            try {
075                InputStream inputStream =
076                    getInputStreamFromResource(Constants.PROPERTIES_RESOURCE);
077                abstractFileConfiguration.load(inputStream);
078            } catch (IllegalArgumentException e) {
079                log.warn(e.getMessage(), e);
080            } catch (ConfigurationException e) {
081                log.warn(e.getMessage(), e);
082            }
083            this.config = abstractFileConfiguration;
084    
085            this.featureModule      = argFeatureModule;
086            this.columnWidths       = new StreamTable();
087            this.columnOrderDefault = new StreamTable();
088            this.columnOrderDefault.put("0", Constants.DESC_NAME);
089            this.columnOrderDefault.put("1", Constants.DESC_MODIFIED);
090            this.columnOrderDefault.put("2", Constants.DESC_IMPORT);
091            this.columnOrderDefault.put("3", Constants.DESC_DELETE);
092            this.sortOrderDefault   = new StreamTable();
093            this.sortOrderDefault.put("0", SortOrder.ASCENDING.toString());
094    
095            instance = this;
096        }
097    
098        public static Preferences getInstance() {
099            Validate.notNull(instance, "You must instantiate Preferences "
100                    + "first before you can access it");
101            if (!instance.initialized) {
102                instance.reload();
103            }
104            return instance;
105        }

108         * Reload context from feature module.
109         */
110        public void reload() {
111            this.featureModule.invoke(Constants.RELOAD_CONTEXT_SUFFIX);
112            instance.initialized = true;
113        }

115        public void setContext(final FeatureModuleContext context) {
116            if (context != null) {
117                this.userPreferences = ((com.moneydance.apps.md.controller.Main)

119            }

122                File file = Common.getPreferencesFile();
123                this.userPreferences = new UserPreferences(file);
124            }
125        }

128            Properties properties   = new Properties();
129            try {
130                InputStream inputStream = Preferences.getInputStreamFromResource(
131                        Constants.LOG4J_PROPERTIES_RESOURCE);
132                properties.load(inputStream);
133            }  catch (IllegalArgumentException e) {
134                e.printStackTrace(System.err);
135            } catch (IOException e) {
136                e.printStackTrace(System.err);
137            }
138            PropertyConfigurator.configure(properties);
139        }

210        public String getColumnName(final int column) {
211            StreamTable streamTable = this.userPreferences.getTableSetting(
212                    Constants.PREF_COLUMN_ORDER,
213                    this.columnOrderDefault);
214            if (this.columnOrder == null) {

220        public void setSortKey(final RowSorter.SortKey sortKey) {
221            StreamTable streamTable = new StreamTable();
222            streamTable.put(sortKey.getColumn(), sortKey.getSortOrder().toString());
223            this.userPreferences.setSetting(
224                    Constants.PREF_SORT_ORDER,
225                    streamTable);
226        }
227    
228        public RowSorter.SortKey getSortKey() {
229            StreamTable streamTable = this.userPreferences.getTableSetting(
230                    Constants.PREF_SORT_ORDER,
231                    null);
232            if (streamTable == null || streamTable.keySet().isEmpty()) {
233                streamTable = this.sortOrderDefault;
234            }
235            Object key        = streamTable.keys().nextElement();
236            int column        = Integer.parseInt(key + "");
237            String sortOrder  = streamTable.getStr(key, null);
238            return new RowSorter.SortKey(column, SortOrder.valueOf(sortOrder));
239        }

244                    this.columnOrderDefault);

246                this.columnOrder = streamTable;
247            }
248            return this.columnOrder.size();

620        private String getTrimmedFilename(final File file) {
621            String filename = file.getName();
622            int length = Math.min(filename.length(), this.getMaxFilenameLength());
623            String trimmedFilename = filename.substring(0, length).trim();
624            if (length < filename.length()) {
625                trimmedFilename = trimmedFilename + "";
626            }
627            return trimmedFilename;
628        }

630        private static InputStream getInputStreamFromResource(
631                final String resource) {
632            ClassLoader cl          = Preferences.class.getClassLoader();
633            InputStream inputStream = cl.getResourceAsStream(resource);
634            Validate.notNull(inputStream,
635                    "Resource " + resource + " was not found.");
636            return inputStream;
637        }
