019    package com.moneydance.modules.features.importlist.util;
020    
021    import java.awt.Color;
022    import java.awt.Font;
023    import java.awt.Toolkit;
024    import java.text.DateFormat;
025    import java.util.Hashtable;
026    import java.util.Locale;
027    import java.util.Observable;
028    
029    import javax.swing.RowSorter;
030    import javax.swing.SortOrder;
031    import javax.swing.UIManager;
032    
033    import org.apache.commons.lang.Validate;
034    
035    import com.moneydance.apps.md.controller.FeatureModuleContext;
036    import com.moneydance.apps.md.controller.UserPreferences;
037    import com.moneydance.util.CustomDateFormat;
038    import com.moneydance.util.StreamTable;
039    
040    /**
041     * This preferences class contains the values the user can control in the
042     * application. It serves as a facade abstracting Moneydance's
043     * <code>UserPreferences</code> (received from the
044     * <code>FeatureModuleContext</code>).
045     *
046     * @author Florian J. Breunig
047     */
048    public final class Preferences extends Observable {
049    
050        private         UserPreferences userPreferences;
051        private final   Settings        settings;
052        private final   StreamTable     columnOrderDefault;
053        private final   StreamTable     sortOrderDefault;
054        private final   StreamTable     columnWidths;
055        private         StreamTable     columnOrder;
056    
057        /**
058         * The constructor must be called exactly once before using the only
059         * instance of this class.
060         */
061        Preferences() {
062            this.settings           = new Settings();
063            this.columnWidths       = new StreamTable();
064            this.columnOrderDefault = new StreamTable();
065            this.columnOrderDefault.put("0", this.settings.getDescName());
066            this.columnOrderDefault.put("1", this.settings.getDescModified());
067            this.columnOrderDefault.put("2", this.settings.getDescImport());
068            this.columnOrderDefault.put("3", this.settings.getDescDelete());
069            this.sortOrderDefault   = new StreamTable();
070            this.sortOrderDefault.put("0", SortOrder.ASCENDING.toString());
071        }
072    
073        public void reload() {
074            this.setChanged();
075            this.notifyObservers(Boolean.TRUE);
076        }
077    
078        public void setContext(final FeatureModuleContext context) {
079            this.userPreferences = ((com.moneydance.apps.md.controller.Main)
080                    context).getPreferences();
081        }
082    
083        private UserPreferences getUserPreferences() {
084            if (this.userPreferences == null) {
085                this.setChanged();
086                this.notifyObservers(Boolean.FALSE);
087                Validate.notNull(this.userPreferences,
088                        "user preferences not initialized");
089            }
090            return this.userPreferences;
091        }
092    
093        public void setAllWritablePreferencesToNull() {
094            this.setBaseDirectory(null);
095            this.getUserPreferences().setSetting(
096                    "importlist.column_widths",
097                    (StreamTable) null);
098            this.setColumnNames(
099                    (Hashtable<String, String>) null);
100            this.getUserPreferences().setSetting(
101                    "importlist.sort_order",
102                    (StreamTable) null);
103        }
104    
105        public String getFullVersion() {
106            final String fullString = this.getUserPreferences().getSetting(
107                    "current_version", "0");
108            return fullString;
109        }
110    
111        public int getMajorVersion() {
112            final String fullString = this.getFullVersion();
113            final int endIndex = Math.min(
114                    this.settings.getLengthOfVersionDigits(),
115                    fullString.length());
116            String substring = fullString.substring(0, endIndex);
117            return Integer.parseInt(substring);
118        }
119    
120        public Locale getLocale() {
121            return this.getUserPreferences().getLocale();
122        }
123    
124        public String getBaseDirectory() {
125            return this.getUserPreferences().getSetting("importlist.import_dir");
126        }
127    
128        public void setBaseDirectory(final String baseDirectory) {
129            this.getUserPreferences().setSetting(
130                    "importlist.import_dir",
131                    baseDirectory);
132        }
133    
134        public boolean useProxy() {
135            return this.getUserPreferences().getBoolSetting(
136                    UserPreferences.NET_USE_PROXY,
137                    false);
138        }
139    
140        public String getProxyHost() {
141            return this.getUserPreferences().getSetting(
142                    UserPreferences.NET_PROXY_HOST);
143        }
144    
145        public int getProxyPort() {
146            return this.getUserPreferences().getIntSetting(
147                    UserPreferences.NET_PROXY_PORT,
148                    0);
149        }
150    
151        public boolean needProxyAuthentication() {
152            return this.getUserPreferences().getBoolSetting(
153                    UserPreferences.NET_AUTH_PROXY,
154                    false);
155        }
156    
157        public String getProxyUsername() {
158            return this.getUserPreferences().getSetting(
159                    UserPreferences.NET_PROXY_USER);
160        }
161    
162        public String getProxyPassword() {
163            return this.getUserPreferences().getSetting(
164                    UserPreferences.NET_PROXY_PASS);
165        }
166    
167        public void setColumnWidths(
168                final int column,
169                final int columnWidth) {
170            this.columnWidths.put(Integer.toString(column), columnWidth);
171            this.getUserPreferences().setSetting(
172                    "importlist.column_widths",
173                    this.columnWidths);
174        }
175    
176        public int getColumnWidths(final int column) {
177            StreamTable streamTable = this.getUserPreferences().getTableSetting(
178                    "importlist.column_widths",
179                    this.columnWidths);
180            return streamTable.getInt(
181                    Integer.toString(column),
182                    this.settings.getColumnWidth());
183        }
184    
185        public void setColumnNames(final Hashtable<String, String> hashtable) {
186            StreamTable streamTable = null;
187            if (hashtable != null) {
188                streamTable = new StreamTable();
189                streamTable.merge(hashtable);
190            }
191            this.getUserPreferences().setSetting(
192                    "importlist.column_order",
193                    streamTable);
194        }
195    
196        public String getColumnName(final int column) {
197            if (this.columnOrder == null) {
198                this.columnOrder = this.getUserPreferences().getTableSetting(
199                        "importlist.column_order",
200                        this.columnOrderDefault);
201            }
202            return this.columnOrder.getStr(Integer.toString(column), null);
203        }
204    
205        public void setSortKey(final RowSorter.SortKey sortKey) {
206            StreamTable streamTable = new StreamTable();
207            streamTable.put(sortKey.getColumn(), sortKey.getSortOrder().toString());
208            this.getUserPreferences().setSetting(
209                    "importlist.sort_order",
210                    streamTable);
211        }
212    
213        public RowSorter.SortKey getSortKey() {
214            StreamTable streamTable = this.getUserPreferences().getTableSetting(
215                    "importlist.sort_order", this.sortOrderDefault);
216            Object key        = streamTable.keys().nextElement();
217            int column        = Integer.parseInt(key.toString());
218            String sortOrder  = streamTable.getStr(key, null);
219            return new RowSorter.SortKey(column, SortOrder.valueOf(sortOrder));
220        }
221    
222        public int getColumnCount() {
223            if (this.columnOrder == null) {
224                this.columnOrder = this.getUserPreferences().getTableSetting(
225                        "importlist.column_order",
226                        this.columnOrderDefault);
227            }
228            return this.columnOrder.size();
229        }
230    
231        public CustomDateFormat getDateFormatter() {
232            return this.getUserPreferences().getShortDateFormatter();
233        }
234    
235        public DateFormat getTimeFormatter() {
236            return DateFormat.getTimeInstance();
237        }
238    
239        /**
240         * @return Preferred width of the file table.
241         */
242        public int getPreferredTableWidth() {
243            return this.settings.getMinimumTableWidth();
244        }
245    
246        /**
247         * @param rowCount The number of rows to display inside the table.
248         * @return Preferred height of the file table.
249         */
250        public int getPreferredTableHeight(final int rowCount) {
251            int fit = this.getHeaderRowHeight()
252                    + rowCount * this.getBodyRowHeight();
253            int min = Math.min(this.getMaximumTableHeight(), fit);
254            return Math.max(this.settings.getMinimumTableHeight(), min);
255        }
256    
257        /**
258         * @return Maximum width of the file table.
259         */
260        public int getMaximumTableWidth() {
261            return (int) Toolkit.getDefaultToolkit().getScreenSize().getWidth()
262                    / (2 + 1);
263        }
264    
265        /**
266         * @return Maximum height of the file table.
267         */
268        public int getMaximumTableHeight() {
269            return (int) Toolkit.getDefaultToolkit().getScreenSize().getHeight()
270                    / (2 + 1);
271        }
272    
273        /**
274         * @return Default foreground color.
275         */
276        public Color getForeground() {
277            int foregroundValue = this.getUserPreferences().getIntSetting(
278                    UserPreferences.GUI_HOMEPG_FG,
279                    this.settings.getColorValueFgDef());
280            return new Color(foregroundValue);
281        }
282    
283        /**
284         * @return Default background color.
285         */
286        public Color getBackground() {
287            int backgroundValue = this.getUserPreferences().getIntSetting(
288                    UserPreferences.GUI_HOMEPG_BG,
289                    this.settings.getColorValueBgDef());
290            return new Color(backgroundValue);
291        }
292    
293        /**
294         * @return Default alternative background color.
295         */
296        public Color getBackgroundAlt() {
297            int backgroundAltValue = this.getUserPreferences().getIntSetting(
298                    "gui.home_alt_bg",
299                    this.settings.getColorValueBgAltDef());
300            return new Color(backgroundAltValue);
301        }
302    
303        public Font getHeaderFont() {
304            final Font baseFont = UIManager.getFont("Label.font");
305            return baseFont.deriveFont(Font.BOLD, baseFont.getSize2D() + 2.0F);
306        }
307    
308        public Font getBodyFont() {
309            return UIManager.getFont("Label.font");
310        }
311    
312        public int getHeaderRowHeight() {
313            return Math.round(
314                    (float) this.settings.getFactorRowHeightHeader()
315                    * this.getBodyFont().getSize2D());
316        }
317    
318        public int getBodyRowHeight() {
319            return Math.round(
320                    (float) this.settings.getSummandRowHeightBody()
321                    + this.getBodyFont().getSize2D());
322        }
323    }
