076        /**
077         * The constructor must be called exactly once before using the only
078         * instance of this class.
079         * @param argFeatureModule The feature module to receive the context from.
080         */
081        public Preferences(final FeatureModule argFeatureModule) {
082            Validate.isTrue(instance == null,
083            "You can instantiate Preferences only once");
084            Validate.notNull(argFeatureModule, "argFeatureModule can't be null");
085    
086            AbstractFileConfiguration abstractFileConfiguration =
087                new PropertiesConfiguration();
088    
089            try {
090                InputStream inputStream =
091                    getInputStreamFromResource(Constants.PROPERTIES_RESOURCE);
092                abstractFileConfiguration.load(inputStream);
093            } catch (IllegalArgumentException e) {
094                log.warn(e.getMessage(), e);
095            } catch (ConfigurationException e) {
096                log.warn(e.getMessage(), e);
097            }
098            this.config = abstractFileConfiguration;
099    
100            this.featureModule      = argFeatureModule;
101            this.columnWidths       = new StreamTable();
102            this.columnOrderDefault = new StreamTable();
103            this.columnOrderDefault.put("0", Constants.DESC_NAME);
104            this.columnOrderDefault.put("1", Constants.DESC_MODIFIED);
105            this.columnOrderDefault.put("2", Constants.DESC_IMPORT);
106            this.columnOrderDefault.put("3", Constants.DESC_DELETE);
107            this.sortOrderDefault   = new StreamTable();
108            this.sortOrderDefault.put("0", SortOrder.ASCENDING.toString());
109    
110            instance = this;
111        }
112    
113        public static Preferences getInstance() {
114            Validate.notNull(instance, "You must instantiate Preferences "
115                    + "first before you can access it");
116            if (!instance.initialized) {
117                instance.reload();
118            }
119            return instance;
120        }

135        public static void loadLoggerConfiguration() {
136            boolean rootIsConfigured =
137                Logger.getRootLogger().getAllAppenders().hasMoreElements();
138            if (rootIsConfigured) {
139                // do not overwrite any existing configurations
140                return;
141            }
142    
143            Properties properties = new Properties();
144            try {
145                InputStream inputStream = Preferences.getInputStreamFromResource(
146                        Constants.LOG4J_PROPERTIES_RESOURCE);
147                properties.load(inputStream);
148            }  catch (IllegalArgumentException e) {
149                e.printStackTrace(System.err);
150            } catch (IOException e) {
151                e.printStackTrace(System.err);
152            }
153            PropertyConfigurator.configure(properties);
154        }

239        public void setSortKey(final RowSorter.SortKey sortKey) {
240            StreamTable streamTable = new StreamTable();
241            streamTable.put(sortKey.getColumn(), sortKey.getSortOrder().toString());
242            this.userPreferences.setSetting(
243                    Constants.PREF_SORT_ORDER,
244                    streamTable);
245        }
246    
247        public RowSorter.SortKey getSortKey() {
248            StreamTable streamTable = this.userPreferences.getTableSetting(
249                    Constants.PREF_SORT_ORDER,
250                    null);
251            if (streamTable == null || streamTable.keySet().isEmpty()) {
252                streamTable = this.sortOrderDefault;
253            }
254            Object key        = streamTable.keys().nextElement();
255            int column        = Integer.parseInt(key + "");
256            String sortOrder  = streamTable.getStr(key, null);
257            return new RowSorter.SortKey(column, SortOrder.valueOf(sortOrder));
258        }

266            }

659        private String getMarkupFilename(final File file) {
660            int length = this.getMessageFilenameLineMaxLength();
661    
662            String filename = file.getName();
663            int numberOfLines = filename.length() / length + 1;
664            String[] substrings = new String[numberOfLines];
665    
666            for (int i = 0; i < numberOfLines; i++) {
667                int start =  i      * length;
668                int end   = (i + 1) * length;
669                substrings[i] = StringUtils.substring(filename, start, end);
670            }
671            return StringUtils.join(substrings, "<br>");
672        }

674        private static InputStream getInputStreamFromResource(
675                final String resource) {
676            ClassLoader cl          = Preferences.class.getClassLoader();
677            InputStream inputStream = cl.getResourceAsStream(resource);
678            Validate.notNull(inputStream,
679                    "Resource " + resource + " was not found.");
680            return inputStream;
681        }
