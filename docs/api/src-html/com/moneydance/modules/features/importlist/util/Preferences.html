018

022import com.moneydance.apps.md.controller.UserPreferences;
023
024import java.awt.Color;
025import java.awt.Font;
026import java.awt.Toolkit;
027import java.text.DateFormat;
028import java.util.Hashtable;
029import java.util.Locale;
030
031import javax.swing.RowSorter;
032import javax.swing.SortOrder;
033import javax.swing.UIManager;
034
035import org.apache.commons.lang3.Validate;
036
037/**
038 * This preferences class contains the values the user can control in the
039 * application. It serves as a facade abstracting Moneydance's
040 * <code>UserPreferences</code> (received from the
041 * <code>FeatureModuleContext</code>).
042 *
043 * @author Florian J. Breunig
044 */
045public final class Preferences {
046
047    private final Settings        settings;
048    private       UserPreferences userPreferences;
049    private final StreamTable     columnOrderDefault;
050    private final StreamTable     sortOrderDefault;
051    private final StreamTable     columnWidths;
052    private       StreamTable     columnOrder;
053
054    /**
055     * The constructor must be called exactly once before using the only
056     * instance of this class.
057     */
058    Preferences() {
059        this.settings           = Helper.INSTANCE.getSettings();
060        this.columnWidths       = new StreamTable();
061        this.columnOrderDefault = new StreamTable();
062        this.columnOrderDefault.put("0", this.settings.getDescName());
063        this.columnOrderDefault.put("1", this.settings.getDescModified());
064        this.columnOrderDefault.put("2", this.settings.getDescImport());
065        this.columnOrderDefault.put("3", this.settings.getDescDelete());
066        this.sortOrderDefault   = new StreamTable();
067        this.sortOrderDefault.put("0", SortOrder.ASCENDING.toString());
068    }
069
070    public void setContext(final FeatureModuleContext context) {
071        this.userPreferences = ((com.moneydance.apps.md.controller.Main)
072                context).getPreferences();
073    }
074
075    private UserPreferences getUserPreferences() {
076        if (this.userPreferences == null) {
077            Helper.INSTANCE.setChanged();
078            Helper.INSTANCE.notifyObservers(Boolean.FALSE);
079            Validate.notNull(this.userPreferences,
080                    "user preferences not initialized");
081        }
082        return this.userPreferences;
083    }
084
085    public void setAllWritablePreferencesToNull() {
086        this.getUserPreferences().setSetting(
087                "importlist.first_run",
088                (String) null);
089        this.setBaseDirectory(null);
090        this.getUserPreferences().setSetting(
091                "importlist.column_widths",
092                (StreamTable) null);
093        this.setColumnNames(
094                (Hashtable<String, String>) null);
095        this.getUserPreferences().setSetting(
096                "importlist.sort_order",
097                (StreamTable) null);
098    }
099
100    public void setFirstRun(final boolean firstRun) {
101        this.getUserPreferences().setSetting(
102                "importlist.first_run",
103                firstRun);
104    }
105
106    public boolean isFirstRun() {
107        return this.getUserPreferences().getBoolSetting(
108                "importlist.first_run",
109                true);
110    }
111
112    public String getFullVersion() {
113        return this.getUserPreferences().getSetting("current_version", "0");
114    }
115
116    public Locale getLocale() {
117        return this.getUserPreferences().getLocale();
118    }
119
120    public String getBaseDirectory() {
121        return this.getUserPreferences().getSetting("importlist.import_dir");
122    }
123
124    public void setBaseDirectory(final String baseDirectory) {
125        this.getUserPreferences().setSetting(
126                "importlist.import_dir",
127                baseDirectory);
128    }
129
130    public boolean hasProxy() {
131        return this.getUserPreferences().getBoolSetting(
132                UserPreferences.NET_USE_PROXY,
133                false);
134    }
135
136    public String getProxyHost() {
137        return this.getUserPreferences().getSetting(
138                UserPreferences.NET_PROXY_HOST);
139    }
140
141    public int getProxyPort() {
142        return this.getUserPreferences().getIntSetting(
143                UserPreferences.NET_PROXY_PORT,
144                0);
145    }
146
147    public boolean hasProxyAuthentication() {
148        return this.getUserPreferences().getBoolSetting(
149                UserPreferences.NET_AUTH_PROXY,
150                false);
151    }
152
153    public String getProxyUsername() {
154        return this.getUserPreferences().getSetting(
155                UserPreferences.NET_PROXY_USER);
156    }
157
158    public String getProxyPassword() {
159        return this.getUserPreferences().getSetting(
160                UserPreferences.NET_PROXY_PASS);
161    }
162
163    public void setColumnWidths(
164            final int column,
165            final int columnWidth) {
166        this.columnWidths.put(Integer.toString(column), columnWidth);
167        this.getUserPreferences().setSetting(
168                "importlist.column_widths",
169                this.columnWidths);
170    }
171
172    public int getColumnWidths(final int column) {
173        StreamTable streamTable = this.getUserPreferences().getTableSetting(
174                "importlist.column_widths",
175                this.columnWidths);
176        return streamTable.getInt(
177                Integer.toString(column),
178                Helper.INSTANCE.getSettings().getColumnWidth());
179    }
180
181    public void setColumnNames(final Hashtable<String, String> hashtable) {
182        StreamTable streamTable = null;
183        if (hashtable != null) {
184            streamTable = new StreamTable();
185            streamTable.merge(hashtable);
186        }
187        this.getUserPreferences().setSetting(
188                "importlist.column_order",
189                streamTable);
190    }
191
192    public String getColumnName(final int column) {
193        if (this.columnOrder == null) {
194            this.columnOrder = this.getUserPreferences().getTableSetting(
195                    "importlist.column_order",
196                    this.columnOrderDefault);
197        }
198        return this.columnOrder.getStr(Integer.toString(column), null);
199    }
200
201    public void setSortKey(final RowSorter.SortKey sortKey) {
202        StreamTable streamTable = new StreamTable();
203        streamTable.put(sortKey.getColumn(), sortKey.getSortOrder().toString());
204        this.getUserPreferences().setSetting(
205                "importlist.sort_order",
206                streamTable);
207    }
208
209    public RowSorter.SortKey getSortKey() {
210        StreamTable streamTable = this.getUserPreferences().getTableSetting(
211                "importlist.sort_order", this.sortOrderDefault);
212        Object key        = streamTable.keys().nextElement();
213        int column        = Integer.parseInt(key.toString());
214        String sortOrder  = streamTable.getStr(key, null);
215        return new RowSorter.SortKey(column, SortOrder.valueOf(sortOrder));
216    }
217
218    public int getColumnCount() {
219        if (this.columnOrder == null) {
220            this.columnOrder = this.getUserPreferences().getTableSetting(
221                    "importlist.column_order",
222                    this.columnOrderDefault);
223        }
224        return this.columnOrder.size();
225    }
226
227    public CustomDateFormat getDateFormatter() {
228        return this.getUserPreferences().getShortDateFormatter();
229    }
230
231    public static DateFormat getTimeFormatter() {
232        return DateFormat.getTimeInstance();
233    }
234
235    /**
236     * @return Preferred width of the file table.
237     */
238    public int getPreferredTableWidth() {
239        return this.settings.getMinimumTableWidth();
240    }
241
242    /**
243     * @param rowCount The number of rows to display inside the table.
244     * @return Preferred height of the file table.
245     */
246    public int getPreferredTableHeight(final int rowCount) {
247        int fit = this.getHeaderRowHeight()
248                + rowCount * this.getBodyRowHeight();
249        int min = Math.min(Preferences.getMaximumTableHeight(), fit);
250        return Math.max(this.settings.getMinimumTableHeight(), min);
251    }
252
253    /**
254     * @return Maximum width of the file table.
255     */
256    public static int getMaximumTableWidth() {
257        return (int) Toolkit.getDefaultToolkit().getScreenSize().getWidth()
258                / (2 + 1);
259    }
260
261    /**
262     * @return Maximum height of the file table.
263     */
264    public static int getMaximumTableHeight() {
265        return (int) Toolkit.getDefaultToolkit().getScreenSize().getHeight()
266                / (2 + 1);
267    }
268
269    public static Font getHeaderFont() {
270        final Font baseFont = UIManager.getFont("Label.font");
271        return baseFont.deriveFont(Font.PLAIN, baseFont.getSize2D() + 2.0F);
272    }
273
274    public static Color getHeaderForeground() {
275        return UIManager.getColor("Label.disabledForeground");
276    }
277
278    public int getHeaderRowHeight() {
279        return Math.round(
280                (float) this.settings.getFactorRowHeightHeader()
281                * Preferences.getBodyFont().getSize2D());
282    }
283
284    public int getBodyRowHeight() {
285        return Math.round(
286                (float) this.settings.getSummandRowHeightBody()
287                + Preferences.getBodyFont().getSize2D());
288    }
289
290    private static Font getBodyFont() {
291        return UIManager.getFont("Label.font");
292    }
293}
